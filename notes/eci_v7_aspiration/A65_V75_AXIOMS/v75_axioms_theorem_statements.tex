\subsection{Hard core (operator-algebra framework)}
\label{sec:axioms-hardcore}

\begin{axiom*}[$H_1$, type-II$_\infty$ crossed product]
The crossed-product von Neumann algebra of observables for the static
patch of de Sitter space, $\widetilde{\mathcal{M}}(\mathcal{D})$ obtained
from the type-III$_1$ algebra $\mathcal{M}(\mathcal{D})$ by gauging the
modular automorphism group of a faithful normal state, is of type
II$_\infty$: it admits no minimal projection and carries a faithful
semifinite normal trace defined up to a positive multiplicative scalar.
\end{axiom*}

\paragraph{Status.}  \textbf{ESTABLISHED} for the static-patch dS
observer algebra (Chandrasekaran--Longo--Penington--Witten 2023,
\cite{CLPW23}, arXiv:2206.10780), corroborated by Kudler-Flam--%
Leutheusser--Satishchandran 2024
(\cite{KLS24}, arXiv:2410.20143).  CONJECTURE for the FRW non-stationary
case (open math.OA question, \cite{eci.tex} limitation \#8).
\textbf{Reference.}  \cite{CLPW23,KLS24}; \cite{eci.tex} \S\,sec:scope
lines 537--560.  \textbf{Falsifier.}  Constructive identification of a
minimal projection in $\widetilde{\mathcal{M}}(\mathcal{D})$ for the
static-patch dS algebra at the Bisognano--Wichmann temperature.

\begin{axiom*}[$H_2$, KMS at $\beta=2\pi$ (Bisognano--Wichmann)]
The state $\omega$ is KMS at the Bisognano--Wichmann inverse temperature
$\beta = 2\pi$ for the modular flow $\sigma^\omega_t$ generated by
Tomita--Takesaki theory; equivalently, the modular Hamiltonian $K_R$
generates a flow of period $2\pi$ on the static-patch algebra and the
correlator analytic-continuation property
$\omega(A\,\sigma^\omega_{i\beta}(B)) = \omega(BA)$ holds for all
$A,B\in\widetilde{\mathcal{M}}(\mathcal{D})$.
\end{axiom*}

\paragraph{Status.}  \textbf{ESTABLISHED} (Tomita--Takesaki theory once
a faithful normal state is fixed; the choice of $\beta = 2\pi$ is the
Bisognano--Wichmann postulate selecting the physical Hartle--Hawking
state).  \textbf{Reference.}  \cite{eci.tex} \S\,sec:scope; standard
QFT-on-curved-spacetime literature.  \textbf{Falsifier.}  Identification
of a faithful normal state on $\widetilde{\mathcal{M}}(\mathcal{D})$
whose modular flow has $\beta\neq 2\pi$ at the static-patch horizon and
admits a physical interpretation incompatible with Hartle--Hawking.

\begin{axiom*}[$H_3$, microlocal functoriality (PH$_k$ axiom A6)]
All constructions are functorial in the locally-covariant sense of
Brunetti--Fredenhagen--Verch (BFV~2003); the persistent-cohomology
functor PH$_k$ used in the dS$_{\rm gen}/d\tau_R$ derivation satisfies
axiom A6 of Kashiwara--Schapira microlocal sheaf theory (existence and
uniqueness of microlocalisation up to canonical isomorphism).
\end{axiom*}

\paragraph{Status.}  \textbf{ESTABLISHED} (BFV 2003 functoriality is
standard; PH$_k$ axiom A6 verification given in the P-KS standalone
paper \texttt{papers\_standalone/}).  \textbf{Reference.}  P-KS paper
``Microlocal sheaves and PH$_k$''; BFV 2003.  \textbf{Falsifier.}  No
direct empirical falsifier (functoriality is a structural compatibility
condition); falsified indirectly if a physical observable computed via
PH$_k$ at one localisation patch fails to agree with its
BFV-functorial-image computation at another patch.

\begin{axiom*}[$H_4$, structural inheritance (vacuous exclusion clause)]
No additional structure is admitted beyond H$_1$--H$_3$: no preferred
vacuum state, no Connes spectral triple, no boundary CFT, no global
extremisation principle.  Equivalently, every dimensionful numerical
observable is determined only modulo the structural-vs-values group
$\mathcal{S} = \mathbb{R}_{>0} \rtimes \mathrm{Out}\bigl(\widetilde
{\mathcal{M}}\bigr)$ (Theorem~1 of \cite{eci.tex}).
\end{axiom*}

\paragraph{Status.}  \textbf{ESTABLISHED} (vacuous exclusion clause; it
is precisely H$_4$ that defines the scope of Theorem~1, the
structure-vs-values gap).  \textbf{Reference.}  \cite{eci.tex}
\S\,sec:scope lines 537--567 (Theorem~1).  \textbf{Falsifier.}  No
direct empirical falsifier; H$_4$ is falsified \emph{a posteriori} by
the very fact that v7.5 must enlarge the axiom system with H$_5'$--%
H$_{11}'$ to recover predictivity.
