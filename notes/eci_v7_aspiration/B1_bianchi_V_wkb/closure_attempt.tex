\section{Mode structure and eigenfunction identification}\label{sec:modes}

The Bianchi~V metric (anisotropic, matter case) is:
\begin{equation}\label{eq:BV-metric}
\mathrm{d}s^2 = -\mathrm{d}t^2 + a_1(t)^2\,\mathrm{d}x^2
  + e^{2x}\bigl[a_2(t)^2\,\mathrm{d}y^2 + a_3(t)^2\,\mathrm{d}z^2\bigr].
\end{equation}
The spatial Lie algebra satisfies $[e_1, e_2] = e_2$, $[e_1, e_3] = e_3$, $[e_2,e_3]=0$
(Bianchi~V structure constants, trace $C^a{}_{1a} = 2$).

\subsection{Friction term and Liouville transform}

With the ansatz $\psi(x,y,z) = R(x)e^{ik_2 y + ik_3 z}$, the spatial Laplacian
$-\Delta_{(3)}$ on the Bianchi~V background gives (sympy-verified):
\begin{equation}\label{eq:friction}
-R''(x) - 2R'(x) + K^2 e^{-2x} R(x) = a_1^2\,\lambda\, R(x),\qquad
K^2 = \Bigl(\tfrac{a_1}{a_2}\Bigr)^2 k_2^2 + \Bigl(\tfrac{a_1}{a_3}\Bigr)^2 k_3^2.
\end{equation}
The friction term $-2R'(x)$ arises from $C^a{}_{1a} = 2$. It is eliminated by the
Liouville transform $R(x) = e^{-x}S(x)$ (sympy-verified):
\begin{equation}\label{eq:liouville}
-S''(x) + K^2 e^{-2x} S(x) = (a_1^2\lambda - 1)\,S(x).
\end{equation}
With $u = Ke^{-x}$ and $\nu^2 = -(a_1^2\lambda - 1)$, this is the modified Bessel
equation $u^2 S'' + uS' - (u^2 + \nu^2)S = 0$. The decaying solution is $K_\nu(u)$.
Setting $\nu = i\rho$ gives $a_1^2\lambda = \rho^2 + 1$, i.e.\ $\lambda = (\rho^2+1)/a_1^2$.

\begin{lemma}[Spatial eigenfunctions, sympy-verified]\label{lem:eigenfunc}
The spatial operator $-\Delta_{(3)}$ on Bianchi~V has eigenfunctions
\[
\psi_{\rho,k_2,k_3}(x,y,z) = N_{\rho,k} \cdot e^{-x} K_{i\rho}(Ke^{-x}) \cdot e^{i(k_2 y + k_3 z)},
\]
with eigenvalue $\lambda(\rho,t) = (\rho^2+1)/a_1(t)^2 \ge 1/a_1^2 > 0$.
The spectral gap $\lambda \ge 1/a_1^2$ (i.e.\ $\rho^2 + 1 \ge 1$) is the $H^3$ spectral gap.
\end{lemma}

\subsection{K--L spectral measure}

The Kontorovich--Lebedev inversion~\cite{Lebedev1965} gives the spectral measure:
\begin{equation}\label{eq:KL-measure}
\mathrm{d}M(\rho,k_2,k_3) = \frac{\rho\sinh(\pi\rho)}{2\pi^4}\,\mathrm{d}\rho\,\mathrm{d}k_2\,\mathrm{d}k_3.
\end{equation}
Note $\mathrm{d}M \to 0$ as $\rho \to 0^+$ (IR cutoff from the $H^3$ spectral gap;
sympy-verified via $\lim_{\rho\to 0}\rho\sinh(\pi\rho) = 0$).

\section{Mehler--Sonine cancellation (leading-order Hadamard, established)}\label{sec:MS}

\begin{proposition}[Mehler--Sonine bound, leading order]\label{prop:MS}
For fixed $u = Ke^{-x} > 0$ and $u' = Ke^{-x'} > 0$ with $u \ne u'$, and
for the K--L spectral measure~\eqref{eq:KL-measure}:
\begin{equation}
|K_{i\rho}(u)|^2 \cdot \rho\sinh(\pi\rho) = O(1) \quad\text{as }\rho\to\infty.
\end{equation}
More precisely, Lebedev~\cite{Lebedev1965} \S6.5 (rigourified by Dunster 2024~\cite{Dunster2024}):
\begin{equation}\label{eq:MS-asymp}
K_{i\rho}(u) = \sqrt{\frac{\pi}{2\rho\sinh(\pi\rho)}} \cdot
  \Bigl[\sin\!\bigl(\rho\ln\tfrac{2\rho}{u} - \rho - \tfrac\pi4\bigr) + E(\rho,u)\Bigr],
\end{equation}
with $|E(\rho,u)| \le C(u)\rho^{-1}$ (\textbf{explicit, computable error bound} from
Dunster 2024~\cite{Dunster2024}).

\emph{Sympy check}: $\rho\sinh(\pi\rho)\cdot\pi/(2\rho\sinh(\pi\rho)) = \pi/2$
(constant, bounded). The oscillation from $\sin^2(\cdots)$ means the integral over
$\rho$ for non-coincident spatial points converges by integration by parts (Riemann--Lebesgue).
\end{proposition}

\section{The missing lemma and proof strategy}\label{sec:lemma}

\begin{lemma}[Olver-BV WKB bound, \emph{to be written up, 4--6~weeks}]\label{lem:WKB}
Let $T^{\mathrm{SLE}}_\rho(t)$ be the SLE minimizer of Theorem~\textup{thm:SLE-BV}
and $T^{\mathrm{WKB}}_\rho(t) = [2\omega(\rho,t)]^{-1/2}\exp(i\int\omega\,\mathrm{d}t)$
the leading WKB vacuum with
\[
\omega(\rho,t)^2 = \frac{\rho^2+1}{a_1(t)^2} + \frac{R(t)}{6}.
\]
Then for $t \in [\delta, T-\delta]$ (bounded away from singularity $t=0$ and future):
\begin{equation}\label{eq:WKB-bound}
|T^{\mathrm{SLE}}_\rho(t) - T^{\mathrm{WKB}}_\rho(t)| \le C(\delta,T)\cdot\rho^{-1}.
\end{equation}
\end{lemma}

\subsection{Proof strategy for Lemma~\ref{lem:WKB}}

\paragraph{STEP A (spatial, $\approx 2$~weeks): Dunster 2024 uniform asymptotics.}

Dunster~(2024)~\cite{Dunster2024} (arXiv:2412.12595, \emph{verified}) derives uniform
Liouville--Green expansions for $K_{i\rho}(z)$ as $\rho\to\infty$, valid in unbounded
complex domains, with explicit error bounds simpler than prior (Olver 1974) results.
The main result gives~\eqref{eq:MS-asymp} with a concrete, computable bound
$|E(\rho,u)| \le C(u)\rho^{-1}$ for fixed $u > 0$.

Applying Dunster's Theorem to our spatial eigenfunctions $K_{i\rho}(Ke^{-x})$
at fixed $x$ (hence fixed $u = Ke^{-x} > 0$ as $\rho\to\infty$) gives the required
spatial bound. This is a direct application of a published 2024 result.

\paragraph{STEP B (temporal, $\approx 2$~weeks): BN23 \S4 Gronwall, verbatim.}

Banerjee--Niedermaier~\cite{BN23} \S4 proves~\eqref{eq:WKB-bound} for Bianchi~I
via a Gronwall inequality on the ODE for $T^{\mathrm{SLE}}_\rho - T^{\mathrm{WKB}}_\rho$.
The argument uses only:
\begin{enumerate}
\item The ODE structure: $\ddot{T} + \omega^2 T = 0$ with Wronskian constraint.
\item The Gronwall estimate: $|T^{\mathrm{SLE}} - T^{\mathrm{WKB}}| \le
  C \int |(\omega^2)' / \omega| \mathrm{d}t \cdot \rho^{-1}$.
\item Compactness: $t \in [\delta, T-\delta]$ keeps $a_1(t)$ away from $0$ and $\infty$.
\end{enumerate}
For Bianchi~V: $\omega(\rho,t)^2 = (\rho^2+1)/a_1(t)^2 + R(t)/6$.
At large $\rho$: $\omega \sim \rho/a_1$ and $|(\omega^2)'/\omega| = O(\rho^{-1})$.
The Gronwall estimate gives $|T^{\mathrm{SLE}} - T^{\mathrm{WKB}}| = O(\rho^{-1})$
verbatim from BN23 \S4, since the argument does not depend on the \emph{spatial}
structure (plane waves vs.\ K--L).

\paragraph{Assembly ($\approx 1$--$2$~weeks).}

With~\eqref{eq:WKB-bound} in hand, the Brum--Them~\cite{BrumThem2013} \S4.2
Sobolev wavefront argument gives $W_{\mathrm{SLE}} - W_0 \in C^\infty(M\times M)$,
hence $\WF(W_{\mathrm{SLE}}) = \WF(W_0) = C^+$ (Radzikowski 1996).

\section{H\texorpdfstring{$^3$}{3} spectral gap and S3 obstruction}\label{sec:obstructions}

\begin{proposition}[S1 voided, S3 active]
For matter Bianchi~V with BKL Kasner attractor $a_i(t) \sim t^{p_i}$ as $t\to 0^+$:
\begin{enumerate}
\item \textbf{S1 voided}: The $H^3$ spectral gap $\lambda_{\min} = 1/a_1^2 > 0$
  removes the zero-mode (constant function) from $L^2(H^3)$.
  The FRW $\log^2$ zero-mode divergence is absent. Sympy-verified: $\lim_{\rho\to 0}\rho\sinh(\pi\rho) = 0$.
\item \textbf{S3 active}: The Kasner constraint $\sum p_i = \sum p_i^2 = 1$
  forces exactly one $p_a < 0$ (contracting direction).
  Example (sympy-verified): $(p_1, p_2, p_3) = (-1/3, 2/3, 2/3)$,
  $\sum = 1$, $\sum^2 = 1$.
  Modes with wave-number $k_a$ in the contracting direction have
  $\omega_k(t) = |k_a|t^{|p_a|} \to 0$ as $t \to 0^+$.
  The smeared two-point function diverges as $\int \mathrm{d}k_a/|k_a| = +\infty$.
  \textbf{S3 alone suffices for T2.}
\end{enumerate}
\end{proposition}
