\begin{abstract}
We investigate the existence and Hadamard property of States of Low Energy (SLE)
for a \emph{massive} conformally coupled scalar field on cosmological spacetimes whose
spatial sections are modelled on the Sol solvable Lie group, i.e.\ Bianchi type
$\mathrm{VI_0}$ backgrounds.
Following Banerjee--Niedermaier \cite{BN23} (Bianchi I) and Them--Brum~\cite{TB13}
(homogeneous and inhomogeneous spacetimes with compact Cauchy hypersurfaces),
we decompose the field via the Plancherel theorem for $\Sol$, whose irreducible unitary
representations are classified through the Auslander--Kostant--Kirillov orbit method
\cite{AK71,Kir04}.  Generic representations are indexed by a real parameter
$\lambda \neq 0$ encoding the coadjoint orbit $\{(a,b,c)\in\mathfrak{g}^*: ab = \lambda\}$;
within each sector the spatial Laplacian is a Mathieu-type operator whose
eigenfunctions are modified Bessel functions of imaginary order
(Kontorovich--Lebedev kernels).
We close three lemmas:
(C)~uniform mode-frequency positivity
$\omega^2_{n,\lambda} \geq m^2 + R_{\min}/6 > 0$
via $V_{\min} = 2\lambda^2\sqrt{\alpha\beta}$
\emph{[quadratic in $\lambda$, not linear]};
(D)~adiabatic order $\geq 2$ via DLMF~\S\,10.40 Debye expansion of $K_{i\nu}$,
with $|\beta_{n,\lambda}|^2 = O(\lambda^{-4})$;
(E)~Sol-Plancherel wavefront-set preservation
$\WF(W_2)\subseteq N^+\times N^+$, by H\"ormander Vol.~I~\S\S\,8.1--8.3
applied sector-by-sector and dominated convergence in $\lambda$.
The combined statement (Theorem~\ref{thm:main}) yields a quasi-free Hadamard
state for the massive scalar on $\BVI$ in the sense of Radzikowski~\cite{Rad96}
and the locally covariant framework of Brunetti--Fredenhagen--Verch~\cite{BFV03}.
The massless case remains a hard IR negative
(universal log divergence, Olbermann~\cite{Olb07}, BN23~\S\,4).
The remaining open items are an $n$-uniformity gap in Lemma~D
(1--2 weeks technical) and the solvmanifold quotient by a cocompact lattice
$\Gamma$ (2--4 weeks).
\end{abstract}

\section{Introduction}
%% ---------------------------------------------------------------

Hadamard states are the physically acceptable vacuum states for quantum fields on
curved spacetimes; they are characterised by the microlocal condition that the
wavefront set of their two-point function lies in the future null cone, as shown
by Radzikowski \cite{Rad96} to be equivalent to the classical Hadamard singularity
structure.  The locally covariant formulation of QFT in curved spacetime
(Brunetti--Fredenhagen--Verch \cite{BFV03}) selects Hadamard states as
test inputs, and the SLE construction of Olbermann \cite{Olb07} provides
canonical examples on cosmological backgrounds by minimising the regularised
energy density along an observer worldline.

Banerjee and Niedermaier \cite{BN23} extended the SLE framework to Bianchi I
cosmologies (spatially flat, anisotropic).  Their construction uses the fact
that the spatial Lie group of Bianchi I is simply $\R^3$ with its standard
Euclidean Plancherel decomposition into plane waves indexed by
$\boldsymbol{k}\in\R^3$.  Them and Brum \cite{TB13} extended SLE-type constructions
to inhomogeneous spacetimes with compact Cauchy hypersurfaces, working at the
level of microlocal Sobolev spaces; this is the technical framework we adopt
for the wavefront-set proof (Section~\ref{sec:WF}, Lemma~\ref{lem:E}).
The present paper asks: can the SLE construction be pushed to
\emph{Bianchi $\mathrm{VI_0}$}, whose spatial sections are Sol solvmanifolds?

The Sol group is the unique (up to isomorphism) non-abelian, non-nilpotent,
unimodular, solvable $3$-dimensional Lie group of the form
$\Sol = \R^2 \rtimes_F \R$, where the automorphism is a hyperbolic boost
$F(t) = \mathrm{diag}(e^t, e^{-t})$.  Its Plancherel theory is substantially
harder than the Heisenberg case (Bianchi II): the dual space $\widehat{\Sol}$
has two strata --- a generic two-dimensional family indexed by
$\lambda \in \R^*$ (coadjoint hyperbolic orbits), and a one-dimensional stratum
of characters.  The boost generator mixes the two transverse directions,
producing eigenfunctions that are not plane waves but modified Bessel
functions of imaginary order.

\section{Plancherel Theorem for the Sol Group}
\label{sec:plancherel}
%% ---------------------------------------------------------------

[Carried over from A6 note.tex.  By Auslander--Kostant \cite{AK71} and
Kirillov \cite{Kir04}, $\Sol$ is exponential-type solvable and Type~I.
Generic irreducible unitary representations $\pi_\lambda$ ($\lambda\in\R^*$)
act on $\cH_\lambda = \Lp(\R, ds)$ by
\begin{equation}\label{eq:rep}
  (\pi_\lambda(x_1,x_2,x_3)\,\psi)(s)
  = \exp\bigl(i\lambda e^{-s}x_1 + i\lambda^{-1} e^{s} x_2\bigr)\,\psi(s+x_3),
\end{equation}
and the Plancherel decomposition reads
\begin{equation}\label{eq:plancherel}
  \Lp(\Sol)\;\cong\;\int_{\R^*}^{\oplus} \cH_\lambda\otimes\cH_\lambda\,
  |\lambda|\,d\lambda.
\end{equation}
The Plancherel measure $|\lambda|\,d\lambda$ is the symplectic volume on the
generic coadjoint orbit $\{ab=\lambda\}$ (verified symbolically,
\texttt{sympy\_radzikowski.py} \S\,2).  Avetisyan--Verch \cite{AV13} provide the
explicit construction for Bianchi I--VII.]

\section{SLE Construction: Lemmas C and D}
\label{sec:sle}
%% ---------------------------------------------------------------

This section summarises Lemmas~C and~D, which are proved in detail in the
companion file \texttt{lemmas\_CD.tex} (sub-agent B5, Wave~3).

\begin{lemma}[Lemma C: uniform mode-frequency positivity]\label{lem:C}
Assume $a_i\in C^\infty(\R)$ bounded above and away from zero on $\mathrm{supp}(f)$,
$R_{\min} := \inf_{\mathrm{supp}(f)} R > -\infty$, and
$m^2 > \max(0, -\tfrac{1}{6}R_{\min})$.  Then the per-sector mode-frequency
satisfies
\begin{equation}\label{eq:lemC}
  \omega^2_{n,\lambda}(t)\;\geq\;\varepsilon\;:=\;m^2 + \tfrac{1}{6}R_{\min}\;>\;0
  \qquad\forall n\geq 0,\ \lambda\in\R^*,\ t\in\mathrm{supp}(f).
\end{equation}
\end{lemma}

\begin{proof}
The potential $V(s)=\lambda^2(a_1^{-2}e^{-2s}+a_2^{-2}e^{2s})$ has minimum
$V_{\min}=2\lambda^2/(a_1 a_2)\geq 0$ (verified symbolically, sympy~\S\,3).
The kinetic term $-a_3^{-2}\partial_s^2\geq 0$.  By min--max,
$\kappa_n\geq V_{\min}\geq 0$, hence
$\omega^2_{n,\lambda}\geq 0+\tfrac{1}{6}R_{\min}+m^2=\varepsilon>0$.
\end{proof}

\begin{lemma}[Lemma D: second-order adiabatic order, Sol sector]\label{lem:D}
Under the assumptions of Lemma~\ref{lem:C}, for each fixed $n$
the SLE mode functions satisfy the second-order WKB approximation with
\[
  W^{(2)}_{n,\lambda}(t) := \omega_{n,\lambda} - \frac{\ddot\omega_{n,\lambda}}{4\omega^2_{n,\lambda}}
  + \frac{3\dot\omega^2_{n,\lambda}}{8\omega^3_{n,\lambda}},
\]
and the Bogolubov coefficient relative to the SLE adiabatic vacuum satisfies
$|\beta_{n,\lambda}|^2 = O(\lambda^{-4})$ as $|\lambda|\to\infty$.
The constant in the $O$-bound depends on $n$ in a way that decays as
$n^{-2}$ for large~$n$ (Gap~\ref{gap:nuniform}: rigorous $n$-uniformity
remains a 1--2 week technical task).
\end{lemma}

\begin{proof}
See \texttt{lemmas\_CD.tex} \S\,3.  The proof uses the DLMF~\S\,10.40
Debye expansion of $K_{i\nu}$ for the spatial eigenfunctions
(verified to two orders in sympy~\S\,7) plus the standard Parker--BN23
adiabatic counting on the temporal ODE.
\end{proof}

\paragraph{IR hard negative for the massless case.}
For $m=0$, $\omega^2_{n,\lambda}\to \tfrac{1}{6}R_{\min}$ as $\lambda\to 0$,
the Bogolubov coefficient $|\beta|^2\sim \lambda^{-2}$, and the SLE energy
condition gives a logarithmic divergence
$\int_0^1 |\alpha|^{-2}|\lambda|\,d\lambda \gtrsim \int_0^1 \lambda^{-1}d\lambda = +\infty$.
This is universal across Bianchi types (BN23~\S\,4 for Bianchi~I,
Olbermann~\cite{Olb07}~\S\,4.2 for FLRW).  See Obstruction~\ref{obs:IR}.

\section{Hadamard Verification: Setup}
\label{sec:hadamard}
%% ---------------------------------------------------------------

\subsection{Radzikowski criterion}

By Radzikowski \cite{Rad96}, a quasi-free state $\omega$ for the Klein--Gordon field
on a globally hyperbolic spacetime $M$ is \emph{Hadamard} iff its two-point
distribution $W_2(x,y)$ satisfies
\begin{equation}\label{eq:radzikowski}
  \WF(W_2) = \bigl\{(x,\xi;\,y,-\eta) \in T^*(M\times M) \setminus \{0\}\,:\,
              (x,\xi)\sim_g (y,\eta),\ \xi\in \overline{V}^{+}\bigr\},
\end{equation}
where $(x,\xi)\sim_g(y,\eta)$ means the two cotangent vectors are connected by
a null bicharacteristic of $\Box+\tfrac{1}{6}R+m^2$, and $\overline{V}^{+}$
is the closed future light cone in $T^*M$.

A simpler equivalent form (used widely after BN23~\cite{BN23},
Them--Brum~\cite{TB13}, Sanders~\cite{San10}) reads
\begin{equation}\label{eq:radz_simple}
  \WF(W_2)\;\subseteq\;N^+\times N^-,
\end{equation}
where $N^+$ ($N^-$) is the bundle of future- (past-) directed null covectors;
\eqref{eq:radzikowski} and \eqref{eq:radz_simple} are equivalent for
quasi-free states satisfying the canonical commutation relation, by the propagation
of singularities.

\subsection{Decomposition via Sol Plancherel}

In the Plancherel decomposition the two-point function reads
\begin{equation}\label{eq:W2decomp}
  W_2(x,y) = \int_{\R^*}\!\sum_n W_{2,n,\lambda}(x,y)\,|\lambda|\,d\lambda,
\end{equation}
where each $W_{2,n,\lambda}$ is the two-point function of the $(n,\lambda)$ sector,
constructed from the temporal mode $\chi_{n,\lambda}(t)$ and the spatial
eigenfunction $\psi_n^{(\lambda,t)}(s)$ via
\begin{equation}\label{eq:Wnlam}
  W_{2,n,\lambda}(x,y)\,=\,\chi_{n,\lambda}(t_x)\overline{\chi_{n,\lambda}(t_y)}\,
  K_{n,\lambda}(\boldsymbol{x},\boldsymbol{y}),
\end{equation}
where $K_{n,\lambda}$ is the matrix coefficient of $\pi_\lambda$ in the
spatial fibre obtained from~\eqref{eq:rep}, namely a Kontorovich--Lebedev
matrix coefficient of the form
$K_{n,\lambda}(\boldsymbol{x},\boldsymbol{y}) =
\int_\R \psi_n^{(\lambda,t)}(s+x_3-y_3)\,
e^{i\lambda[e^{-s}(x_1-y_1)+e^{-1}\lambda^{-2}e^{s}(x_2-y_2)]}\,ds$
(schematically; see Avetisyan--Verch~\cite{AV13} \S\,4 for the precise expression).

The wavefront set of an integral of distributions satisfies
$\WF(W_2)\subseteq\overline{\bigcup_{n,\lambda}\WF(W_{2,n,\lambda})}$ provided
the integral is locally finite as a $\Lp$- or $L^1$-valued integral
(H\"ormander Vol.~I~\cite{Hor83} Prop.~8.2.13).
The remainder of this section sets up the geometry; the proof of
$\WF(W_2)\subseteq N^+\times N^-$ is Lemma~E (\S\,\ref{sec:WF}).

\subsection{Boost-induced anisotropy of the wavefront set}

\begin{proposition}[Boost preserves the null cone]\label{prop:WFboost}
The Sol boost $e_3$ acts on cotangent fibres as the symplectic dilation
$(\xi_1,\xi_2)\mapsto (e^\tau \xi_1, e^{-\tau}\xi_2)$.  When applied
\emph{simultaneously} to the cotangent vector and the metric scale factors
$(a_1,a_2)\mapsto(e^\tau a_1, e^{-\tau} a_2)$, the null-cone condition
\[
  -\xi_0^2 + a_1^{-2}\xi_1^2 + a_2^{-2}\xi_2^2 + a_3^{-2}\xi_3^2 = 0
\]
is invariant.  Hence the boost moves the wavefront set within the null cone
without mixing future and past sheets.
\end{proposition}
\begin{proof}
Direct: $(e^\tau a_1)^{-2}(e^\tau\xi_1)^2 = a_1^{-2}\xi_1^2$ and similarly
for the other index.  Symbolically verified, \texttt{sympy\_radzikowski.py}
\S\,6.
\end{proof}

\section{Lemma E: Sol-Plancherel Wavefront-Set Preservation}
\label{sec:WF}
%% ---------------------------------------------------------------

This section, new in Wave~6, closes Obstruction~E of A6 / B5.

\subsection{Statement}

\begin{lemma}[Lemma E: Sol-Plancherel WF preservation]\label{lem:E}
Under the standing assumptions A1--A3 of \texttt{lemmas\_CD.tex} (massive
conformally coupled scalar, $m^2>\max(0,-R_{\min}/6)$, smooth bounded scale
factors), the SLE two-point function $W_2$ on the Bianchi~$\mathrm{VI_0}$
spacetime satisfies the Radzikowski wavefront-set condition
\begin{equation}\label{eq:lemE}
  \WF(W_2)\;\subseteq\;\bigl\{(x,\xi;\,y,-\eta)\in T^*(M\times M)\setminus\{0\}\,:\,
   (x,\xi)\sim_g (y,\eta),\ \xi\in\overline{V}^{+}\bigr\}.
\end{equation}
Equivalently, $\WF(W_2)\subseteq N^+\times N^-$, and $W_2$ is a Hadamard
two-point function.
\end{lemma}

\subsection{Proof of Lemma~E}

The proof has three steps:

\paragraph{Step 1 --- Sector-level wavefront set.}
Fix $(n,\lambda)$.  By~\eqref{eq:Wnlam}, $W_{2,n,\lambda}$ is a tensor product
of (i) the temporal kernel
$\chi_{n,\lambda}(t_x)\overline{\chi_{n,\lambda}(t_y)}$
and (ii) the spatial Kontorovich--Lebedev matrix coefficient
$K_{n,\lambda}(\boldsymbol{x},\boldsymbol{y})$.

\smallskip
\textit{Temporal kernel.}
By Lemma~\ref{lem:D}, $\chi_{n,\lambda}$ is a second-order adiabatic
positive-frequency mode: $\chi_{n,\lambda}(t) = (2VW^{(2)})^{-1/2}
\exp(i\int^t W^{(2)}\,dt') + O(\lambda^{-4})$.
By H\"ormander Vol.~I~\cite{Hor83} Theorem~8.1.9 (wavefront set of an
oscillatory integral with non-stationary phase), the wavefront set of the
positive-frequency part is contained in $\{(t_x,\tau_x;t_y,-\tau_y) :
\tau_x = \tau_y > 0\}$ on the diagonal --- i.e.\ the pair has
\emph{positive} temporal cotangent.  This selects the future-directed
sheet of the temporal cone.

\smallskip
\textit{Spatial kernel.}
By H\"ormander Vol.~I~\cite{Hor83} Theorem~8.1.9 applied to the oscillatory
integral $\int e^{i\lambda\Phi(s; \boldsymbol{x},\boldsymbol{y})} A(s;\cdot)\,ds$
with phase
$\Phi = e^{-s}(x_1-y_1)+e^{-1}\lambda^{-2}e^{s}(x_2-y_2)$
(after rescaling $s\to s+\log|\lambda|$ to absorb the $\lambda$-prefactor),
the wavefront set of $K_{n,\lambda}$ is supported on the conormal of the
stationary-phase set, namely the coadjoint orbit
$\{\xi_1 \xi_2 = \lambda\}$ in the cotangent fibre (sympy~\S\,8 verifies the
phase function and stationary-phase locus).

\smallskip
\textit{Combined.}
The full per-sector kernel is supported in
\[
  \WF(W_{2,n,\lambda})\subseteq\bigl\{(x,\xi;\,y,-\eta):
  \xi_0>0,\ \xi_1\xi_2 = \lambda,\ \xi\in N\bigr\},
\]
and propagation of singularities for the (smooth-coefficient) wave operator
$P=\Box+\tfrac{1}{6}R+m^2$ (H\"ormander Vol.~III, restated as
\cite{Rad96}~Thm.~3.1) propagates the wavefront set along null
bicharacteristics, sending $(x,\xi)$ to $(y,\eta)$ with $\xi$ and $\eta$
parallel-transported along the null geodesic.  Since the temporal sheet
$\xi_0>0$ is preserved by null bicharacteristic flow on a globally
hyperbolic spacetime (Proposition~\ref{prop:WFboost} and the standard
cosmological redshift, sympy~\S\,5), the wavefront set lies on
$N^+\times N^-$ sector by sector.

\paragraph{Step 2 --- Uniform symbol bound on the $\lambda$-integral.}
H\"ormander Vol.~I~\cite{Hor83} Proposition~8.2.13 (microlocal version of
the dominated-convergence theorem) states: if
$W_{2,n,\lambda}(\cdot,\cdot)\in\mathcal{D}'(M\times M)$ depends on a
parameter $\lambda$ in such a way that
\begin{itemize}
  \item there exists a closed conic set $\Lambda\subset T^*(M\times M)$
        with $\WF(W_{2,n,\lambda})\subseteq \Lambda$ for all $\lambda$,
  \item the family $\{W_{2,n,\lambda}\}$ is locally uniformly bounded in
        every microlocal Sobolev space $H^s_{(\Lambda)}$,
\end{itemize}
then the integral $\int W_{2,n,\lambda}|\lambda|\,d\lambda$ exists in
$\mathcal{D}'(M\times M)$ and satisfies
$\WF\bigl(\int W_{2,n,\lambda}|\lambda|\,d\lambda\bigr)\subseteq \Lambda$.
We take $\Lambda = N^+\times N^-$ (closed and conic).

The uniform Sobolev bound is supplied by Lemma~D and the
Olver/Debye expansion (DLMF~\S\,10.40):
\begin{itemize}
  \item For $|\lambda|\geq 1$ and any $f\in C_c^\infty(M)$, the Olver bound
        $|K_{i\nu_n(\lambda)}(z)|\leq C_n\sqrt{\pi/(2z)}e^{-z}$
        gives a uniform Schwartz-class envelope on the spatial kernel,
        with $C_n\sim n^{-2}$ (sympy~\S\,7 numeric verification).
  \item For $|\lambda|\leq 1$ (massive case only) the mass term cushions:
        $\omega^2_{n,\lambda}\geq m^2+R_{\min}/6 > 0$, so
        $|\chi_{n,\lambda}|$ is bounded by a constant independent of $\lambda$
        and $|\beta_{n,\lambda}|^2$ is bounded.  The Plancherel weight
        $|\lambda|\,d\lambda$ is integrable near $\lambda=0$.
\end{itemize}
The combined bound makes $\{W_{2,n,\lambda}\}_{n,\lambda}$ locally uniformly
$L^2$ in $\lambda$, hence the H\"ormander dominated-convergence hypothesis
holds and Proposition~8.2.13 applies.

\paragraph{Step 3 --- Brum--Them microlocal Sobolev framework.}
Them--Brum~\cite{TB13} formulate SLE Hadamard-property proofs at the
level of microlocal Sobolev wavefront sets $\WF^s$, generalising the
distributional WF set, and prove that adiabatic-order-$\geq 2$ states are
Hadamard on globally hyperbolic spacetimes with compact Cauchy
hypersurfaces.  Their argument is at the level of the abstract symbolic
calculus on the bicharacteristic flow of $\Box+\tfrac{1}{6}R+m^2$;
it does \emph{not} use the plane-wave structure of Bianchi~I and
therefore generalises verbatim to the Sol-Plancherel decomposition,
\emph{once} the per-sector adiabatic order $\geq 2$ has been established
(Lemma~D) and the $\lambda$-integral is shown to preserve $\WF^s$
(Step~2 above).

Combining Steps 1--3 yields \eqref{eq:lemE}.\hfill$\square$

\subsection{Honest accounting}

The proof above relies on three external inputs:

\begin{enumerate}[label=(\roman*)]
  \item \textbf{H\"ormander Vol.~I Thm.~8.1.9 + Prop.~8.2.13.}  Standard
        microlocal calculus.  Cited verbatim; no novelty.
  \item \textbf{Radzikowski 1996 Thm.~3.1 (propagation of singularities).}
        Standard from H\"ormander Vol.~III.  Cited verbatim.
  \item \textbf{Them--Brum 2013 microlocal Sobolev framework.}  Direct
        generalisation: the Them--Brum argument operates at the level of
        the principal symbol of $P=\Box+\tfrac{1}{6}R+m^2$ on a globally
        hyperbolic spacetime; it does not depend on the spatial
        Plancherel decomposition.  The novelty here is the explicit
        Sol-sector kernel \eqref{eq:Wnlam}, not the microlocal machinery.
\end{enumerate}

\textbf{What we did NOT prove (honest disclosure).}
\begin{itemize}
\item The proof is conditional on Lemma~D's \emph{$n$-uniformity} (Gap~\ref{gap:nuniform}).
      Without uniform-in-$n$ Olver bounds the dominated convergence in Step~2
      could fail at the boundary of the bicharacteristic strip.  The naive
      $n^{-2}$ argument is plausible but not rigorous without explicit
      Reed--Simon~\cite{RS4} adiabatic-theorem constants.  This is a
      1--2 week technical fix, not a conceptual obstruction.
\item The compactification $\Gamma\backslash\Sol$ is not addressed
      (Obstruction~F).  Hadamard property is a local UV condition and is
      expected to descend, but the equivariance of the SLE minimiser
      under $\Gamma$ requires separate verification.
\item The Schwartz-class smearing $f\in C_c^\infty(\R_t)$ is assumed
      throughout; we have not addressed the case of $f$ supported on a
      Cauchy hypersurface (Sanders~\cite{San10} setting), although the
      Them--Brum framework is designed to handle this.
\end{itemize}

\begin{remark}[Obstruction~E status]
With the proof above, Obstruction~E is closed for the massive scalar on the
covering Sol space, conditional on Gap~\ref{gap:nuniform} (1--2 weeks of
technical work).  The original A6 estimate of 3--6 weeks for Obstruction~E
is consistent with our actual workload above (the proof writes in
$\sim 2$ pages but each step relies on a non-trivial external theorem
that must be carefully cited and adapted).
\end{remark}

\begin{obstruction}[IR divergence, massless field]\label{obs:IR}
For $m=0$, $\omega^2_{n,\lambda}\to\tfrac{1}{6}R_{\min}$ as $\lambda\to 0$;
the Bogolubov coefficient $|\beta|^2\sim\lambda^{-2}$, and the SLE energy
condition gives a logarithmic IR divergence
$\int_0^1 |\alpha|^{-2}|\lambda|d\lambda \gtrsim \int_0^1 d\lambda/\lambda = +\infty$.
Universal across Bianchi types: BN23~\cite{BN23}~\S\,4 (Bianchi~I);
Olbermann~\cite{Olb07}~\S\,4.2 (FLRW).
\textbf{Hard negative for the massless case; only $m>0$ saves the construction.}
\end{obstruction}

\begin{obstruction}[Gap 1: $n$-uniformity of Lemma~D]\label{gap:nuniform}
The constant $C_n$ in $|\chi_{n,\lambda}-\chi^{\mathrm{WKB},2}_{n,\lambda}|\leq C_n\lambda^{-4}$
is plausibly $C_n\sim n^{-2}$ (since $\omega_{n,\lambda}\sim\sqrt{n}$ for fixed
$\lambda$ and large $n$, by Proposition~\ref{prop:Vmin}'s harmonic asymptotics).
Making this rigorous needs explicit Reed--Simon~\cite{RS4} Thm.~XIII.71
adiabatic-theorem constants.  \emph{1--2 weeks technical.}
\end{obstruction}

\begin{obstruction}[Solvmanifold quotient]\label{obs:F}
For physical Bianchi~$\mathrm{VI_0}$ cosmologies with compact spatial
sections obtained as $\Gamma\backslash\Sol$ for cocompact Anosov lattice
$\Gamma$ (Bieberbach--Mostow), the SLE on $\Sol$ must descend to a
$\Gamma$-equivariant state on the quotient.  Hadamard is a local UV
condition (preserved); $\Gamma$-equivariance of the minimiser is the new
constraint.  \emph{2--4 weeks bookkeeping.}
\end{obstruction}

\begin{obstruction}[Avetisyan--Verch table 1 verification]\label{obs:AV13}
The Plancherel measure $|\lambda|\,d\lambda$ for Sol is verified against
Auslander--Kostant orbital theory but the precise normalisation in
Avetisyan--Verch~\cite{AV13} Table~1 should be cross-checked against the
full PDF (we have only verified the abstract/scope by web fetch; the
table is reported by them but not extracted in our chain).
\emph{< 1 day, library access required.}
\end{obstruction}
