\noindent\textbf{Setting.}  Continuing from Lemma 3.3, let
$(\mathcal{M}_{\rm FRW},g_{\rm FRW})$ be a spatially flat FLRW spacetime
in conformally flat coordinates,
$g_{\rm FRW}=a(\eta)^2(-d\eta^2+d\mathbf{x}^2)$, $d=4$.  Let $\Phi$ be
the conformally coupled massless real scalar ($\xi=1/6$) in the
conformal vacuum $\Omega_{\rm conf}\in\mathcal{H}_{\rm FRW}$,
intertwined with the Minkowski vacuum $\Omega_M$ by the unitary $U$ of
Lemma~3.3.  Let $D_R(\eta_c,\mathbf{x}_c)$ be the conformal diamond of
conformal radius $R$, $B_R(\mathbf{x}_c)$ its central spacelike slice,
and $K_{\rm FRW}$ its modular Hamiltonian (Proposition~3.4
of~\cite{Lemma33}).

\medskip
\noindent\textbf{Generalised entropy on type II$_\infty$.}  Following
Chandrasekaran--Longo--Penington--Witten (CLPW) for de~Sitter
\cite{CLPW2022,Witten2021},
\begin{equation}
   S_{\rm gen}[D_R]
   \;=\;
   \langle K_{\rm FRW}\rangle_\psi \;+\; S_{\rm vN}(\rho_\psi)
   \;+\; \Delta_{\rm anom}[D_R],
   \label{eq:Sgen}
\end{equation}
where the third term is the c-number anomaly contribution computed
below; the area-law divergence absorbed into the type~II$_\infty$ trace
renormalisation contributes only an additive state-independent constant
(\cite[\S5.4]{CLPW2022}, \cite[\S2]{Witten2021}).

\medskip
\noindent\textbf{The trace anomaly.}  The conformal-anomaly Wick
polynomial $A\equiv g^{\mu\nu}\langle T_{\mu\nu}\rangle_{\rm ren}$ for
the conformally coupled massless scalar in $d=4$ is, in any locally
covariant Hadamard prescription \cite{HollandsWald2001,HollandsWald2014},
\begin{equation}
   A \;=\; \frac{1}{2880\,\pi^2}\Bigl[
        R_{\mu\nu\rho\sigma}R^{\mu\nu\rho\sigma}
        - R_{\mu\nu}R^{\mu\nu}
        + \Box R \Bigr]
   \;\;\equiv\;\;
   \frac{1}{16\pi^2}\bigl[c\,W^2-a\,E_4\bigr] +
        \frac{1}{2880\,\pi^2}\,\Box R,
   \label{eq:trace-anomaly}
\end{equation}
with $c=1/120$, $a=1/360$, $W^2=R_{\mu\nu\rho\sigma}R^{\mu\nu\rho\sigma}-2R_{\mu\nu}R^{\mu\nu}+\frac13R^2$
the Weyl invariant, and $E_4=R_{\mu\nu\rho\sigma}R^{\mu\nu\rho\sigma}-4R_{\mu\nu}R^{\mu\nu}+R^2$
the Euler density.  The two presentations of \eqref{eq:trace-anomaly}
differ by a $\Box R$ counterterm:
sympy verification of $({A_{\rm BD}}-{A_{(a,c)}})/\Box R=1/(2880\pi^2)$
on conformally flat FRW is performed in
\texttt{sympy\_anomaly\_v2.py}~\S(II).
Wald's axiomatic prescription \cite{Wald1978} (Axiom~5: no derivatives
of the metric beyond second order in $T_{\mu\nu}$) sets the $\Box R$
coefficient to zero, leaving only the $(a,c)$-anomaly (the
``Riegert'' form).  Christensen's point-splitting prescription
\cite{Christensen1976} retains the finite $\Box R$ counterterm with
coefficient $1/(2880\pi^2)$.  Both prescriptions agree on radiation
era ($\Box R=0$) and on the unique part of $A$ that contributes to
$S_{\rm gen}$ via Lemma~\ref{lem:34} below.

On the spatially flat FRW background, $W^2=0$, so the IR-finite,
regulator-independent piece is
\begin{equation}
   A_{\rm FRW}
   \;=\; \frac{1}{2880\,\pi^2}
        \Bigl[R_{\mu\nu}R^{\mu\nu}-\tfrac13R^2+\Box R\Bigr]
        \;=\; \frac{-E_4}{5760\,\pi^2}+\frac{\Box R}{2880\pi^2}\,,
   \label{eq:A-FRW}
\end{equation}
where the conformally flat identity $R_{\mu\nu\rho\sigma}R^{\mu\nu\rho\sigma}=2R_{\mu\nu}R^{\mu\nu}-\frac13R^2$
(used here at $d=4$, Birrell--Davies~\cite[eq.~(6.123)]{BirrellDavies})
is sympy-verified by \texttt{sympy\_anomaly\_v2.py}~\S(I).

\smallskip
\noindent\textbf{Curvature invariants of FRW (sympy-exact).}
In the conformally flat coordinates $(η,\mathbf{x})$ with mostly-plus
signature,
\begin{align*}
   R&=\frac{6\,a''}{a^3}-\frac{6(a')^2}{a^4}\,\equiv\,\frac{6\,a''}{a^3}\,
       \text{when written via }H\equiv a'/a:\;\;R=\tfrac{6}{a^2}(\dot H+2H^2),\\
   R_{\mu\nu}R^{\mu\nu}&=\frac{1}{a^4}\Bigl[\bigl(-3\tfrac{a''}{a}+3H^2\bigr)^2
        +3\bigl(\tfrac{a''}{a}+H^2\bigr)^2\Bigr],\\
   R_{\mu\nu\rho\sigma}R^{\mu\nu\rho\sigma}
        &=2\,R_{\mu\nu}R^{\mu\nu}-\tfrac13R^2\quad(\text{Weyl}=0),\\
   \Box R&=-\frac{1}{a^2}\bigl(R''+2HR'\bigr).
\end{align*}
Specialisations (\texttt{sympy\_anomaly\_v2.py}~\S(III)):
\begin{equation}
\begin{array}{l|cccc}
& R & R_{\mu\nu}R^{\mu\nu} & R_{\mu\nu\rho\sigma}R^{\mu\nu\rho\sigma}
& \Box R\\\hline
\text{Radiation }a=\eta &
0 & 12/\eta^{8} & 24/\eta^{8} & 0\\
\text{Matter }a=\eta^{2} &
12/\eta^{6} & 144/\eta^{12} & 240/\eta^{12} & -216/\eta^{12}
\end{array}
\label{eq:curv-table}
\end{equation}
hence $A_{\rm FRW}|_{\rm rad}=1/(240\pi^2\eta^8)>0$ and
$A_{\rm FRW}|_{\rm mat}=-1/(24\pi^2\eta^{12})<0$.

\medskip
\noindent\textbf{Anomaly-determined stress tensor.}  Bunch--Davies
\cite{BunchDavies1977} (specifically PRSA A~\textbf{356}, p.~569) derive
the locally covariant, conformally vacuum-aligned, conserved stress
tensor on FRW(k=0) by a covariant point-splitting prescription.  In
the conformally flat frame, write
\[
   \langle T_{00}\rangle_\psi=a^2\rho(\eta),\qquad
   \langle T_{ii}\rangle_\psi=a^2 p(\eta)\quad(\text{isotropy}),
\]
so that the trace identity $g^{\mu\nu}\langle T_{\mu\nu}\rangle=A$
becomes $-\rho+3p=A$ and the conformal-frame conservation
$\nabla^\mu\langle T_{\mu\nu}\rangle=0$ reduces to
$\rho'+3H(\rho+p)=0$.  Eliminating $p$ yields the first-order linear
inhomogeneous ODE
\begin{equation}
   \rho'(\eta)+4H(\eta)\rho(\eta)=-H(\eta)A_{\rm FRW}(\eta),
   \label{eq:rho-ODE}
\end{equation}
whose general solution is, by the integrating factor $a^4$,
\begin{equation}
   a^4(\eta)\,\rho(\eta)
   \;=\;-\!\int^{\eta}\!a^3(\tilde\eta)\,a'(\tilde\eta)\,A_{\rm FRW}(\tilde\eta)\,
        d\tilde\eta
   \;+\;C_0.
   \label{eq:rho-int-factor}
\end{equation}
The Bunch--Davies/Hadamard prescription fixes $C_0=0$ (no asymptotic
Casimir residue at $\eta\to\infty$; this is the unique choice
compatible with the locally covariant, microlocal-spectrum-condition
Hadamard parametrix \cite{HollandsWald2001}).  The resulting
$\rho_{\rm anom}\equiv\rho|_{C_0=0}$ defines the \emph{anomaly-driven,
state-independent} part $\langle T_{00}\rangle_{\rm anom}=
a^2\rho_{\rm anom}$.

\smallskip
Closed-form values (sympy-verified, \texttt{sympy\_anomaly\_v2.py}~\S(IV)):
\begin{equation}
\boxed{\;
\begin{array}{l|c}
  & \langle T_{00}(\eta)\rangle_{\rm anom}\\\hline
  \text{Radiation }a=\eta
  & +\dfrac{1}{960\,\pi^2\,\eta^{6}}\\[6pt]
  \text{Matter }a=\eta^{2}
  & -\dfrac{1}{48\,\pi^2\,\eta^{8}}
\end{array}\;}
\label{eq:T00-anom-eras}
\end{equation}
Conservation $\rho_{\rm anom}'+3H(\rho_{\rm anom}+p_{\rm anom})=0$
and the trace identity $-\rho_{\rm anom}+3p_{\rm anom}=A_{\rm FRW}$
both check identically on each era and on the cross-test profile
$a=\eta^{3/2}$ (\texttt{sympy\_anomaly\_v2.py}~\S(IV)).

\begin{lemma}[Anomaly bookkeeping for $S_{\rm gen}[D_R]$, $d=4$]
\label{lem:34}
For the conformally coupled massless real scalar in the conformal
vacuum on the spatially flat FLRW background, the state-independent
anomaly contribution to the generalised entropy~\eqref{eq:Sgen} of the
FRW diamond $D_R(\eta_c,\mathbf{x}_c)$ is
\begin{equation}
\boxed{\,
   \Delta_{\rm anom}[D_R]
   \;=\;
   \frac{8\pi^2}{15}\,a(\eta_c)^2\,R^4\,
   \langle T_{00}(\eta_c)\rangle_{\rm anom}
\,}
   \label{eq:lem34}
\end{equation}
where $\langle T_{00}\rangle_{\rm anom}$ is the unique conserved,
state-independent, trace-anomaly-saturating $(0,0)$-component of
$\langle T^{\rm FRW}_{\mu\nu}\rangle$ defined by
\eqref{eq:rho-int-factor} with $C_0=0$.  In the two physically
relevant epochs:
\begin{itemize}
\item Radiation era $a(\eta)=\eta$:
\begin{equation}
   \Delta_{\rm anom}[D_R]\;=\;+\dfrac{R^4}{1800\,\eta_c^{4}}\quad(\text{POSITIVE}).
   \label{eq:lem34-rad}
\end{equation}
\item Matter era $a(\eta)=\eta^2$:
\begin{equation}
   \Delta_{\rm anom}[D_R]\;=\;-\dfrac{R^4}{90\,\eta_c^{4}}\quad(\text{NEGATIVE}).
   \label{eq:lem34-mat}
\end{equation}
\end{itemize}
The $R^4 a(\eta_c)^2$-scaling is uniform across cosmological epochs;
the $\eta_c^{-4}$ dependence in the explicit formulas is also uniform
(both eras), reflecting the dimensional scaling
$\langle T_{00}\rangle_{\rm anom}\propto a^{-2}\,(\text{curvature})^2$
combined with the $a(\eta_c)^2$ measure.  The sign flips between
radiation (positive) and matter (negative); this is the
$(a,c)$-anomaly fingerprint, dominated by $\Box R$ on matter (where
$E_4<0$ overcomes $R_{\mu\nu}R^{\mu\nu}$).
\end{lemma}

\begin{proof}
\emph{Step 1.\ Locality and homogeneity of the anomaly piece.}  By
Hollands--Wald \cite[\S5]{HollandsWald2001}, the renormalised stress
tensor on a globally hyperbolic spacetime decomposes as
\begin{equation*}
   \langle T^{\rm FRW}_{\mu\nu}\rangle_\psi
   \;=\;
   T^{\rm anom}_{\mu\nu}[g_{\rm FRW}] \;+\; t^{\psi}_{\mu\nu},
\end{equation*}
with $T^{\rm anom}_{\mu\nu}$ a local Wick polynomial in the metric
(state-independent) and $t^{\psi}_{\mu\nu}$ the state-dependent residue,
finite at any spacelike point in any Hadamard state.  By spatial
homogeneity of $g_{\rm FRW}$, $T^{\rm anom}_{\mu\nu}$ depends only on
$\eta$.  By isotropy, $T^{\rm anom}_{0i}=0$ and the spatial part is a
multiple of $\delta_{ij}$.  Trace and conservation conditions
\begin{align*}
   g^{\mu\nu}T^{\rm anom}_{\mu\nu}&=A_{\rm FRW},
   &\nabla^\mu T^{\rm anom}_{\mu\nu}&=0,
\end{align*}
together with $T^{\rm anom}_{00}=a^2\rho_{\rm anom}$, yield the
ODE~\eqref{eq:rho-ODE}.  The integrating-factor solution
\eqref{eq:rho-int-factor} with the Bunch--Davies/Hadamard normalisation
$C_0=0$ is unique among locally covariant Wick polynomials with
no Casimir residue.

\smallskip
\emph{Step 2.\ Insertion into Proposition~3.4 of \cite{Lemma33}.}
Lemma~3.3 + Hislop--Longo \cite{HislopLongo1982} give
\begin{equation*}
   K_{\rm FRW}
   \;=\; 2\pi\!\int_{B_R(\mathbf{x}_c)}\!d^3x\;
        \frac{R^2-|\mathbf{x}-\mathbf{x}_c|^2}{2R}\;
        a(\eta_c)^2\;\mathcal{T}^{\rm FRW}_{00}(\eta_c,\mathbf{x})
        + c_0,
\end{equation*}
with $\mathcal{T}^{\rm FRW}_{00}$ the locally covariant, conformally
vacuum-aligned, conserved stress operator (Lemma~3.3).  Taking the
expectation in any state $\psi$ and decomposing
$\mathcal{T}^{\rm FRW}_{00}=T^{\rm anom}_{00}+t^{\psi}_{00}$, the
anomaly contribution $\langle T^{\rm anom}_{00}(\eta_c)\rangle$ is
spatially constant (Step~1) and factors out:
\begin{align*}
   \Delta_{\rm anom}[D_R]
   &=2\pi\,a(\eta_c)^2\,
        \langle T^{\rm anom}_{00}(\eta_c)\rangle\!
        \int_{B_R(\mathbf{x}_c)}\!d^3 x\,\frac{R^2-|\mathbf{x}-\mathbf{x}_c|^2}{2R}.
\end{align*}
The kernel integral is sympy-verified
(\texttt{sympy\_anomaly\_v2.py}~\S(V)):
\begin{equation}
   \int_{B_R}\!d^3x\,\frac{R^2-r^2}{2R}
   =\int_0^R\!4\pi r^2\,\frac{R^2-r^2}{2R}\,dr
   =\frac{4\pi R^4}{15}.
   \label{eq:kernel-integral}
\end{equation}
Substituting yields the boxed formula~\eqref{eq:lem34}.

\smallskip
\emph{Step 3.\ Specialisations.}  Substituting the table~\eqref{eq:T00-anom-eras}:
\begin{align*}
   \Delta_{\rm anom}[D_R]\Big|_{a=\eta}
   &=\tfrac{8\pi^2}{15}\,\eta_c^2\,R^4\cdot
        \tfrac{1}{960\,\pi^2\,\eta_c^{6}}
   =\tfrac{R^4}{1800\,\eta_c^{4}},\\
   \Delta_{\rm anom}[D_R]\Big|_{a=\eta^2}
   &=\tfrac{8\pi^2}{15}\,\eta_c^4\,R^4\cdot
        \!\left(-\tfrac{1}{48\,\pi^2\,\eta_c^{8}}\right)
   =-\tfrac{R^4}{90\,\eta_c^{4}}.
\end{align*}
Numerical sanity check at $(\eta_c,R)=(1,1/2)$: $\Delta_{\rm anom}^{\rm rad}
=3.4722\!\cdot\!10^{-5}$, $\Delta_{\rm anom}^{\rm mat}=-6.9444\!\cdot\!10^{-4}$,
mpmath@200dps cross-check (\texttt{sympy\_anomaly\_v2.py}~\S(VII)).

\smallskip
\emph{Step 4.\ Affiliation to the type II$_\infty$ algebra.}
$\Delta_{\rm anom}$ is a c-number; it lies in the centre
$\mathbb{C}\!\cdot\!\mathbf{1}$ of every von Neumann algebra and is
therefore trivially affiliated to the type II$_\infty$ factor of the
carve-out.  The crossed-product trace renormalisation absorbs the local
$\Box R$ counterterm into the additive constant of $\log\rho_\psi$
(cf.~\cite[\S2.3]{Witten2021}), so the physical generalised entropy is
$\Delta_{\rm anom}$-shifted by a finite, regulator-independent c-number.
\end{proof}

\begin{remark}[Comparison with the area-law]\label{rem:area}
The proper area of $\partial B_R$ in the FRW metric is
$|\partial B_R|_{\rm phys}=4\pi R^2 a(\eta_c)^2$.  The CHM
area-law term \cite{CasiniHuertaMyers2011} is
$|\partial B_R|_{\rm phys}/(4G_N)=\pi R^2 a(\eta_c)^2/G_N$.  The
dimensional ratio is (sympy-verified, \texttt{sympy\_anomaly\_v2.py}~\S(VIII)):
\begin{align*}
   \frac{\Delta_{\rm anom}^{\rm rad}}{|\partial B_R|_{\rm phys}/(4G_N)}
   &=\frac{G_N R^2}{1800\,\pi\,\eta_c^{6}},\\
   \frac{\Delta_{\rm anom}^{\rm mat}}{|\partial B_R|_{\rm phys}/(4G_N)}
   &=-\frac{G_N R^2}{90\,\pi\,\eta_c^{8}}.
\end{align*}
At late times ($\eta_c\to\infty$) the ratio vanishes polynomially,
consistent with asymptotic Minkowskian behaviour of the conformal
vacuum.

The Solodukhin universal logarithmic correction
\cite[eq.~(5.55)]{Solodukhin2011} is
\begin{equation}
   S_{\rm EE}^{\rm univ-log}
   \;=\;-\frac{1}{360}\,\log\frac{|\partial B_R|_{\rm phys}}{4\pi\epsilon^2}
   \;=\;-\frac{a_{\rm anomaly}}{1}\,\log(R/\epsilon)+\text{const},
   \label{eq:solod-log}
\end{equation}
with $a_{\rm anomaly}=1/360$ for the conformally coupled scalar.  This
finite, regulator-independent log-coefficient is \emph{not} absorbed by
the type II$_\infty$ trace renormalisation; it is part of $S_{\rm gen}$
on equal footing with $\Delta_{\rm anom}$.  We carry it as an additive
contribution to~\eqref{eq:Sgen} but do not enlarge the scope of
Lemma~\ref{lem:34} beyond the $\Delta_{\rm anom}$ piece.
\end{remark}

\begin{remark}[Sign discussion: radiation vs matter]\label{rem:sign}
The sign flip $\Delta_{\rm anom}^{\rm rad}>0>\Delta_{\rm anom}^{\rm mat}$
reflects the structure of $A_{\rm FRW}$: on radiation,
$\Box R=0$ so the trace is $A_{\rm FRW}=R_{\mu\nu}R^{\mu\nu}/(2880\pi^2)
=R_{\mu\nu\rho\sigma}R^{\mu\nu\rho\sigma}/(5760\pi^2)>0$ (Kretschmann
positive on Lorentzian conformally flat).  On matter,
$\Box R=-216/\eta^{12}<0$ dominates: $A_{\rm FRW}=-1/(24\pi^2\eta^{12})<0$.
The integrating-factor solution \eqref{eq:rho-int-factor} preserves
the sign of $A_{\rm FRW}$ to leading order, so $\langle T_{00}\rangle_{\rm anom}$
inherits the sign.

The B1-wave statement of this lemma quoted both eras as positive with
$\eta_c^{-6}$ and $\eta_c^{-8}$ scaling.  The B1 result was based on a
dimensionally inconsistent ansatz
$\langle T_{00}\rangle_{\rm anom}\propto R_{\mu\nu}R^{\mu\nu}$ which
neglects the $1/a^2$ redshift factor implicit in conformal-frame
conservation.  The corrected derivation here yields uniform
$\eta_c^{-4}$ scaling and an exact rational coefficient
$1/(1800\,\eta_c^4)$ (radiation) and $-1/(90\,\eta_c^4)$ (matter).
\end{remark}

\begin{corollary}[Type II$_\infty$ structure preservation]\label{cor:typeII}
$\Delta_{\rm anom}[D_R]$ is a c-number-valued contribution to
$S_{\rm gen}$.  It commutes with every observable and lies in the
centre $\mathbb{C}\!\cdot\!\mathbf{1}$ of the local algebra
$\mathcal{A}(D_R)$.  Therefore the carve-out's type II$_\infty$
structure (Witten 2021 \cite{Witten2021}, CLPW 2022 \cite{CLPW2022})
is unaffected: the crossed-product algebra
$\mathcal{A}(D_R)\rtimes\mathbb{R}$ remains a type II$_\infty$
factor, and $S_{\rm gen}$ is well-defined modulo the standard
type II$_\infty$ additive constant.
\end{corollary}

\begin{openq}[Q3.5: Wald--Zoupas extrinsic-curvature residual]
The boundary $\partial D_R$ is a Killing horizon of the conformally
mapped Minkowski metric but \emph{not} of $g_{\rm FRW}$ (which carries
only spatial-translation/rotation Killing vectors; no boost generator
on $D_R$ in FRW bulk).  The CHM modular flow of Lemma~3.3 is
conformally covariant, so the leading $\Delta_{\rm anom}$ piece is
correctly~\eqref{eq:lem34}; however, the Wald--Zoupas symplectic-
current prescription \cite{WaldZoupas2000} adds a sub-leading
extrinsic-curvature contribution from the FRW-vs-Minkowski mismatch on
$\partial D_R$.  Order-of-magnitude estimate:
\[
   \frac{\delta(\Delta_{\rm anom})}{\Delta_{\rm anom}}\;\sim\; R\,K_{\rm extr}\;\sim\; R\,H(\eta_c)
   \;=\;\begin{cases}\dfrac{R}{\eta_c}, & a=\eta\;\text{(radiation)},\\
        \dfrac{2R}{\eta_c}, & a=\eta^2\;\text{(matter)}.\end{cases}
\]
For $R\ll\eta_c$ (small diamond compared to the cosmological time
scale), the correction is sub-leading and Lemma~\ref{lem:34} is
sufficient for the carve-out paper's submission scope.  A full
Wald--Zoupas computation (with explicit symplectic flux through
$\partial D_R$ in FRW bulk) is deferred to a companion note.
\end{openq}
