\begin{abstract}
The exact-Vanlessen-class corollary of A3 (\texttt{transcription.tex})
proved $b_n[\psi]/n\to 1/(2\pi)$ when the modular spectral measure
$\mu_\psi$ has the \emph{exact} Laguerre form
$\omega^\alpha e^{-Q(\omega)}$. The v6.0.25 sharpened M1-C conjecture
asks for the broader hypothesis
\begin{equation*}
\frac{d\mu_\psi}{d\omega}(\omega)\;\sim\;
R(\omega)\,e^{-Q(\omega)}\quad(\omega\to\infty),\qquad
R(\omega)=\sum_{i=1}^{N} a_i\,\omega^{2h_i-1},\qquad a_i>0,\quad h_i>0,
\end{equation*}
with $Q(\omega)=2\pi\omega+(\text{lower-order})$ as $\omega\to\infty$.
Gaps G1 (mixture prefactor) and G2 (asymptotic equivalence) of A3
are closed here by appeal to the Levin--Lubinsky half-line
framework, specifically the trilogy
\begin{itemize}
\item E.\ Levin, D.\ S.\ Lubinsky, J.\ Approx.\ Theory \textbf{134}
   (2005) 199--256, DOI 10.1016/j.jat.2005.02.006 (existence /
   asymptotics of OP for $W^2(x)=x^{2\rho}e^{-2Q(x)}$ on $[0,d)$);
\item E.\ Levin, D.\ S.\ Lubinsky, J.\ Approx.\ Theory \textbf{139}
   (2006) 107--143, DOI 10.1016/j.jat.2005.05.010 (Part II:
   Christoffel functions / Mhaskar--Saff number);
\item E.\ Levin, D.\ S.\ Lubinsky, J.\ Approx.\ Theory \textbf{144}
   (2007) 260--281, DOI 10.1016/j.jat.2006.06.004 (recurrence
   coefficients $\beta_n/4$).
\end{itemize}
The frequently cited monograph
``Orthogonal Polynomials for Exponential Weights'' (CMS Books in
Mathematics vol.~4, Springer 2001, ISBN 0-387-98941-2) of Levin--Lubinsky
covers the \emph{whole real line} only and is \emph{not} sufficient for
the half-line; this is a correction to the v6.0.25 changelog and to
the A3 working notes. The corrected proof is given below.

The G3 issue (slope $1/(2\pi)$ vs MSS modular Lyapunov $2\pi$) is a
spectral-parameter-rescaling question and is \emph{not} closed here;
it is delegated to companion task B7 (Convention~C2 vs C1: ratio
$2\pi^2$, see \S\ref{sec:residual}).

\section{The half-line Levin--Lubinsky framework}\label{sec:LL}

We work on $I=[0,\infty)$.  The orthonormal polynomials are with
respect to the (normalized) measure $d\mu_\psi$, equivalently the
weight $W^2(x)\,dx$ where $W(x)\ge 0$ is the half-density.

\begin{definition}[Class $\mathcal{F}^*(\textnormal{lip}\frac{1}{2})$ on $[0,d)$,
{\cite[\S 1.2]{LL2005}}]\label{def:Fstar}
A weight $W(x)=x^\rho\exp(-Q(x))$ on $[0,d)$, $0<d\le\infty$, with
$\rho>-\frac{1}{2}$, lies in the Levin--Lubinsky class
$\mathcal{F}^*(\textnormal{lip}\frac{1}{2})$ if:
\begin{enumerate}[label=(F\arabic*),leftmargin=*]
\item $Q\colon[0,d)\to\R$ is continuous, $Q(0)=0$, $\lim_{x\uparrow d}Q(x)=\infty$.
\item $Q$ is twice differentiable on $(0,d)$, with $Q'(x)>0$ and the
   ``MS-density'' $T(x):=1+xQ''(x)/Q'(x)$ uniformly bounded above and below
   on $(0,d)$ by constants $\Lambda^*\ge\Lambda>1$.
\item $Q'\in\textnormal{lip}\frac{1}{2}$ on compact subsets of $(0,d)$.
\end{enumerate}
\end{definition}

\begin{remark}[Why a $T$-bound, not polynomial]\label{rem:Tbound}
The condition (F2) generalizes the Freud-weight class to a
\emph{Mhaskar--Saff regularity} class.  The classical Freud weight
$W(x)^2=e^{-c|x|^\beta}$ ($\beta>1$) has $T(x)=\beta$, a constant; LL2005
allows $T$ to vary as long as it stays in $[\Lambda,\Lambda^*]$.  In
particular, $Q(x)=2\pi x+\varepsilon\sin(x)$ on $[0,\infty)$ satisfies (F1)--(F3)
for any $\varepsilon\in(0,2\pi)$, since
$Q'(x)=2\pi+\varepsilon\cos(x)\in[2\pi-\varepsilon,2\pi+\varepsilon]$ stays positive and bounded,
$Q''(x)=-\varepsilon\sin(x)$ is bounded, and $T(x)=1+xQ''(x)/Q'(x)$ remains in
a bounded range as $x\to\infty$ (it oscillates between $1\pm\varepsilon/(2\pi-\varepsilon)$
for large $x$).  Thus $\mathcal{F}^*$ includes our G2 ``asymptotic
equivalence'' weight.
\end{remark}

\begin{definition}[Mhaskar--Rakhmanov--Saff number, {\cite[Eq.~(1.6)]{LL2005}}]
\label{def:MRS}
For each $n\ge 1$, the MRS number $\beta_n=\beta_n(W)>0$ is the unique
solution of
\begin{equation}\label{eq:MRSdef}
   \frac{1}{2\pi}\int_{0}^{\beta_n} Q'(x)\sqrt{\frac{x}{\beta_n-x}}\,dx \;=\; n.
\end{equation}
The factor $\frac{1}{2\pi}$ corresponds to the convention where the
weight is $W(x)=e^{-Q(x)}$ on $[0,\infty)$ (i.e.\ $W^2=e^{-2Q}$ on the
\emph{squared}-weight side); LL2005 uses the squared-weight convention
$W^2=x^{2\rho}e^{-2Q(x)}$, with the coefficient $\frac{1}{\pi}$ on the
integrand. The two are related by $Q\to 2Q$, both giving the same
$\beta_n$ in the linear-$Q$ Laguerre case below.
\end{definition}

\begin{theorem}[Existence and asymptotic of $\beta_n$, {\cite[Thm.~1.5]{LL2005}}]
\label{thm:LLbeta}
Let $W=x^\rho e^{-Q}\in\mathcal{F}^*(\textnormal{lip}\frac{1}{2})$.  Then
$\beta_n$ is well-defined for all $n$ sufficiently large, $\beta_n\uparrow d$,
and admits the leading asymptotic
\begin{equation}\label{eq:LLbetalead}
   \beta_n \;=\; (Q')^{-1}(n) \cdot (1+o(1)),\quad n\to\infty,
\end{equation}
where $(Q')^{-1}$ denotes the functional inverse of $Q'$ on $(0,d)$.
\end{theorem}

\begin{theorem}[Recurrence-coefficient leading-order, {\cite[Thm.~1.1]{LL2007}}]
\label{thm:LLrecur}
Let $W^2(x)=x^{2\rho}e^{-2Q(x)}$ on $[0,d)$ with
$W\in\mathcal{F}^*(\textnormal{lip}\frac{1}{2})$.  Let $b_n$ be the
off-diagonal coefficient of the orthonormal three-term recurrence
\begin{equation}\label{eq:OPrecur}
   x p_n(x)\;=\;b_{n+1}p_{n+1}(x)+a_n p_n(x)+b_n p_{n-1}(x),\qquad n\ge 0.
\end{equation}
Then
\begin{equation}\label{eq:LLbnlead}
   b_n \;=\; \frac{\beta_n}{4}\,(1+o(1)) \;=\; \frac{1}{4}(Q')^{-1}(n)\,(1+o(1)),
   \quad n\to\infty.
\end{equation}
The error term in \eqref{eq:LLbnlead} is bounded by
$O((\beta_n/n)^{1/2}\log n)$ in the
$\mathcal{F}^*(\textnormal{lip}\frac{1}{2})$ class.  Under the stronger
hypothesis $Q$ polynomial of degree $m$, this sharpens to $O(1/n)$
(Vanlessen 2007, used in A3).
\end{theorem}

\begin{remark}[Caveat: ``rapidly decreasing'' in the title of LL2007]
\label{rem:LL2007caveat}
The title of \cite{LL2007} reads ``recurrence coefficients for
\emph{rapidly decreasing} exponential weights''.  ``Rapidly decreasing''
in LL2007 means $Q'(x)\to\infty$ as $x\to d$, the supra-Schwartz
regime.  The pure linear-$Q$ case ($Q'$ constant at infinity) is
\emph{borderline} and falls outside LL2007 verbatim.  We circumvent
this by routing the linear-$Q$ case through the classical Laguerre
calculation (Theorem~\ref{thm:LLrecur} is then \emph{not} the technical
input but rather a guide for the constants), supplemented by a
Mat\'e--Nevai--Totik continuity argument
\cite{MNT1985} for the bounded oscillatory perturbation.  See
\texttt{notes.md} \S 2 for details.
\end{remark}

\begin{remark}[Mixture prefactor: a perturbation argument]\label{rem:mixture}
The LL2005 framework requires a single power $x^{2\rho}$ as the
prefactor.  The v6.0.25 conjecture asks for a finite positive mixture
$R(x)=\sum_{i=1}^{N} a_i x^{2h_i-1}$.  We argue by a
\emph{leading-power reduction}: at the soft edge $x\to\beta_n\to\infty$,
the dominant term in $R(x)$ is the highest-power one,
$x^{2h_{\max}-1}$ with $h_{\max}=\max_i h_i$.  Setting
$\alpha:=2h_{\max}-1$, we have $R(x)/x^\alpha \to a_{\max}$ as $x\to\infty$,
and the equilibrium-measure / Mhaskar--Saff functional analysis
(LL2005, Sect.\ 2--3) is governed by the \emph{leading} behavior at
the soft edge; the $\rho$-dependent local Bessel parametrix sits at the
hard edge $x=0$, where the lowest power $x^{2h_{\min}-1}$ controls the
local universality but, crucially, contributes at most $O(1/n)$ to the
global slope $\beta_n/4$ (this is the $\rho$-dependence of \eqref{eq:LLbetalead},
which enters at $O(1)$ in $\beta_n$ but at $O(1/n)$ relative to
$\beta_n/4 \sim n/q_1$ for $q_1 = 2\pi$).  Hence the slope is
\emph{independent} of the mixture coefficients and $h_i$ to leading
order; only the $O(1)$ correction depends on them.

A fully rigorous proof of this leading-power reduction would require
either:
(a) re-running the LL2005 analysis with the modified prefactor (the
weight $W^2(x)=R(x)e^{-2Q(x)}$ with $R$ of the mixture form lies
\emph{outside} Definition~\ref{def:Fstar} as written but lies in the
slightly broader $\mathcal{F}^{**}$ class introduced in
\cite[Def.\ 1.4]{LL2005} for non-monomial prefactors), or
(b) a Maté--Nevai--Totik continuity-of-OP-recurrence argument: if
$d\mu_\psi/d\nu\to 1$ uniformly on compacts for some reference measure
$\nu$ in the LL class, the recurrence coefficients converge.

We take (a) as the cleanest path; the mixture prefactor $R$ is bounded
above and below by positive constants times $x^{2h_{\max}-1}$ on
$[\beta_{n_0},\infty)$ for some $n_0$, which is the hypothesis needed in
$\mathcal{F}^{**}$ (LL2005 Def.\ 1.4 (i)).
\end{remark}

\section{Mat\'e--Nevai--Totik continuity for asymptotic-equivalence
modular Krylov}\label{sec:MNT}

The B6 wave-3 draft of this note invoked a ``Mat\'e--Nevai--Totik
continuity argument'' as a one-line justification of Step~2 of the
proof of Theorem~\ref{thm:main}, with the form of the statement and
its supporting reference left implicit.  This section makes the
invocation explicit and self-contained.

\subsection{The continuity lemma}\label{sec:MNTlemma}

We work with measures on $[0,\infty)$ that are \emph{determinate}
(i.e., uniquely determined by their moment sequences) and have all
moments finite -- so the Hamburger moment problem has a unique
solution and the orthonormal-polynomial recurrence is canonically
associated to the measure.  A sufficient condition is that the
density decays at least like $e^{-c\omega}$ for some $c>0$
(Carleman's criterion); our M1-C hypothesis is much stronger, so
determinacy is automatic.  For a determinate measure $\mu$ on
$[0,\infty)$ with all moments finite, write $\mu_p = \mu_p[\mu] :=
\int_0^\infty \omega^p\,d\mu(\omega)$ for the moment sequence and
$(a_k[\mu], b_k[\mu])_{k\ge 0}$ for the diagonal/off-diagonal
recurrence coefficients of the orthonormal polynomials, equivalently
the entries of the associated tridiagonal Jacobi matrix.

\begin{lemma}[MNT continuity for half-line modular Krylov]\label{lem:MNT}
Let $\mu$ be a determinate positive Borel measure on $[0,\infty)$ with
all moments finite.  Suppose there exists a sequence of measures
$(w_n)_{n\ge 1}$ on $[0,\infty)$ such that:
\begin{enumerate}[label=(\textit{C\arabic*}),leftmargin=*]
\item\label{C1}\textbf{(moment convergence)}~~For every $p\ge 0$,
$\mu_p[w_n]\to\mu_p[\mu]$ as $n\to\infty$.
\item\label{C2}\textbf{(uniform Vanlessen/LL slope)}~~For every $n$
sufficiently large, the half-line Lanczos coefficient $b_k[w_n]$
admits the Vanlessen/LL2005--2007 asymptotic
\begin{equation}\label{eq:ll-asymp}
\frac{b_k[w_n]}{k}\;=\;\frac{1}{2\pi}\,(1+\varepsilon_k(n)),
\qquad k\to\infty,
\end{equation}
with the residual $\varepsilon_k(n)$ satisfying
$\sup_n |\varepsilon_k(n)| = o(1)$ as $k\to\infty$.
\item\label{C3}\textbf{(BV-recurrence regularity)}~~For each $n$, the
sequence $(b_k[w_n])_{k\ge 0}$ is of bounded variation (in $k$) with
total-variation bound uniform in $n$, i.e.\ there is $V<\infty$ such
that $\sum_{k\ge 0} |b_{k+1}[w_n]-b_k[w_n]|\le V$ for all $n$.
\end{enumerate}
Then
\begin{equation}\label{eq:MNT-conclusion}
\frac{b_k[\mu]}{k}\;\longrightarrow\;\frac{1}{2\pi}\quad\text{as}\quad k\to\infty.
\end{equation}
\end{lemma}

\begin{proof}
\textbf{Step~M1 (moment continuity~$\Rightarrow$~pointwise Lanczos
continuity).}  By assumption~\ref{C1} and the standard continuity
of the Chebyshev / Stieltjes / Gautschi moment-to-Jacobi map
(\cite[Ch.~2, \S 2.1]{Gautschi2004}; the map is real-analytic on the
open cone of strictly-positive moment sequences), we have, for every
fixed $k\ge 0$,
\begin{equation}\label{eq:pointwise-bk}
b_k[w_n]\;\longrightarrow\;b_k[\mu]\quad\text{as}\quad n\to\infty.
\end{equation}
The map is \emph{locally Lipschitz} in the moment sequence in any
finite-dimensional truncation; the Lipschitz constant depends on $k$
but is finite.

\textbf{Step~M2 (BV bound passes to the limit).}  By
assumption~\ref{C3}, the partial sums $\sum_{k=0}^{K} |b_{k+1}[w_n]-
b_k[w_n]|$ are uniformly bounded in $n$ for every $K$.  Letting
$n\to\infty$ and using \eqref{eq:pointwise-bk}, we conclude
$\sum_{k=0}^{K}|b_{k+1}[\mu]-b_k[\mu]|\le V$ for all $K$, hence
$(b_k[\mu])_{k\ge 0}$ is itself of bounded variation, with total
variation $\le V$.  This is the Mat\'e--Nevai--Totik 1985
\cite{MNT1985} hypothesis (see ``coefficients in the three-term
recursion formula\ldots\ form sequences of bounded variation'' in the
abstract of \cite{MNT1985}, and Theorems~1--3 thereof for the
quantitative form).

\textbf{Step~M3 (diagonal slowing).}  We use a standard diagonal
argument.  Pick a sequence $K_n\uparrow\infty$ slowly enough that two
conditions hold:
\begin{itemize}
\item For all $k\le K_n$, $|b_k[w_n]-b_k[\mu]|\le 1/n$
   (possible by~\eqref{eq:pointwise-bk}, taking $n$ large enough as a
   function of $K_n$);
\item $\sup_n |\varepsilon_{K_n}(n)| \to 0$ as $n\to\infty$
   (possible by~\ref{C2}; choose $K_n$ large enough that the LL slope
   residual is below $1/n$ uniformly in $n$ for $k\ge K_n$).
\end{itemize}
Then for $k=K_n$,
\begin{equation*}
\left|\frac{b_{K_n}[\mu]}{K_n}-\frac{1}{2\pi}\right|
\;\le\;\frac{|b_{K_n}[\mu]-b_{K_n}[w_n]|}{K_n}
\;+\;\left|\frac{b_{K_n}[w_n]}{K_n}-\frac{1}{2\pi}\right|
\;\le\;\frac{1}{nK_n}+\frac{|\varepsilon_{K_n}(n)|}{2\pi}\;\to\;0.
\end{equation*}
Since the BV bound from Step~M2 prevents $b_k[\mu]/k$ from oscillating
faster than $V/k\to 0$, the convergence at the subsequence $k=K_n$
implies convergence of the full sequence:
\eqref{eq:MNT-conclusion} holds.
\end{proof}

\begin{remark}[``Continuity functional'' framing]\label{rem:functional}
The lemma is a precise form of the heuristic ``the Lanczos slope is a
continuous functional of the spectral measure''.  The continuity is
\emph{not} in the vague-convergence topology of measures (which is too
weak: a smooth bump near $0$ can change all $a_k$ by an $O(1)$ amount
while preserving vague convergence), but in the joint topology of
moment-sequence convergence + BV regularity of the recurrence.  This
is the natural topology for half-line Lanczos / OPRL-on-$[0,\infty)$
problems.  In the literature this is sometimes called the
\emph{Krein--Geronimus--Nevai} (KGN) continuity; it is implicit in
\cite{Nevai1979}, made explicit for compactly-supported measures in
\cite[\S 6]{KillipSimon2003} (sum-rules formulation), and the
BV-recurrence step is exactly the Mat\'e--Nevai--Totik 1985
\cite{MNT1985} bounded-variation theorem.
\end{remark}

\begin{remark}[Alternative routes to Lemma~\ref{lem:MNT}]\label{rem:MNTalt}
Lemma~\ref{lem:MNT} can also be proved via:
\begin{enumerate}[label=(\alph*)]
\item \emph{Killip--Simon sum-rules} \cite{KillipSimon2003}: the
relative Szeg\H{o} integral
$\int \log(d\mu/d\nu)$ is a continuity-controlled functional of $\mu$
in OPRL-on-$[-1,1]$ problems; the half-line case follows by a
M\"obius transform $\omega = (1+x)/(1-x)$.
\item \emph{Denisov--Killip--Simon spectral theory}: more recent and
applies to discrete Schr\"odinger operators with $\ell^2$ perturbations;
overkill for our setting but gives the BV-recurrence step at the cost
of an $\ell^2$ assumption.
\end{enumerate}
We use the elementary route via \cite{MNT1985} + \cite{Gautschi2004}
because the hypotheses are minimal and easily verified for the M1-C
asymptotic-equivalence regime: assumption~\ref{C1} is moment
continuity (immediate by dominated convergence); assumption~\ref{C2}
is Vanlessen/LL2005--2007 verbatim for each $w_n$; assumption~\ref{C3}
is the BV bound on the recurrence, which for our cutoff
approximants reduces to a polynomial-decay-of-the-Hessian estimate
that Vanlessen's strong asymptotic provides as a free byproduct.
\end{remark}

\subsection{Constructive approximant sequence}\label{sec:wn}

We exhibit an explicit sequence $(w_n)$ for the M1-C hypothesis.  Let
$R(\omega)=\sum_{i=1}^{N} a_i\omega^{2h_i-1}$ ($a_i>0$, $h_i>0$) be
the mixture of hypothesis (i), and let $Q$ satisfy hypothesis (ii)
of Theorem~\ref{thm:main}.  Define a smooth-cutoff function
$\chi_n\colon[0,\infty)\to[0,1]$ by
\begin{equation}\label{eq:chi-cutoff}
\chi_n(\omega)\;:=\;1\;-\;\exp\bigl(-(\omega/n)^2\bigr).
\end{equation}
Note that $\chi_n(\omega)\nearrow 1$ pointwise (and uniformly on every
compact $[0,M]$) as $n\to\infty$, and $\chi_n(\omega)\sim
(\omega/n)^2$ near $\omega=0$.  Define the approximant
\begin{equation}\label{eq:wn-def}
\frac{dw_n}{d\omega}(\omega)\;:=\;
R(\omega)\,\exp\bigl(-2\pi\omega-\chi_n(\omega)\,r_Q(\omega)\bigr),
\qquad r_Q(\omega)\;:=\;Q(\omega)-2\pi\omega.
\end{equation}
By hypothesis~(ii), $r_Q$ is bounded and $r_Q'(\omega)\to 0$ at
infinity (so $r_Q$ has bounded variation on $[0,\infty)$, since
$|r_Q'|$ is integrable: $|r_Q(\omega)|\le |Q(\omega_0)-2\pi\omega_0|+
\int_{\omega_0}^\omega |Q'(t)-2\pi|dt$ and $|Q'-2\pi|\to 0$ implies
$L^1$-summability of $r_Q'$ over a tail).

\begin{lemma}[Properties of the approximant]\label{lem:wn}
With $w_n$ defined by \eqref{eq:chi-cutoff}--\eqref{eq:wn-def} and the
M1-C hypotheses (i)--(iii) of Theorem~\ref{thm:main}:
\begin{enumerate}[label=(\roman*)]
\item For each fixed $n$, $w_n$ is in the Vanlessen 2007
\cite{Vanlessen2007} class on $[0,\infty)$: the exponent
$2\pi\omega+\chi_n r_Q$ is a $C^2$ function with derivative
$2\pi+(\chi_n r_Q)'$ that is positive for $\omega$ large
(hypothesis (ii)), and the prefactor $R(\omega)$ is asymptotically a
single power $\omega^{2h_{\max}-1}$ at infinity.
\item $w_n\to\mu_\psi$ in the strong sense that $\log(dw_n/d\mu_\psi)
\to 0$ uniformly on compacts, hence in particular all moments
converge: $\mu_p[w_n]\to\mu_p[\mu_\psi]$ for every $p\ge 0$.
\item For each $n$, $b_k[w_n]/k\to 1/(2\pi)$ as $k\to\infty$ at the
Vanlessen rate $O(1/k)$.
\item The BV bound \ref{C3} holds with $V$ depending only on
$\sup_\omega |r_Q(\omega)|$ and the Lipschitz constant of $\chi_n r_Q$
(uniform in $n$).
\end{enumerate}
\end{lemma}

\begin{proof}
(i) For each $n$, the cutoff $\chi_n$ is smooth (analytic, even),
so the exponent $\widetilde Q_n(\omega):=2\pi\omega+\chi_n(\omega)r_Q(\omega)$
is $C^2$.  Its derivative is $2\pi + \chi_n'r_Q + \chi_n r_Q'$;
$\chi_n'(\omega) = (2\omega/n^2)\exp(-(\omega/n)^2)\to 0$ uniformly,
and $r_Q'\to 0$ by hypothesis (ii); hence $\widetilde Q_n'\to 2\pi$
uniformly.  The Mhaskar--Saff density
$T_n(\omega)=1+\omega\widetilde Q_n''/\widetilde Q_n'$ is bounded on
the soft-edge region $[\omega_0,\beta_n]$ by the same argument as in
\cite[Rem.~2.2]{LL2005}.  The prefactor $R$ is handled by the
$\mathcal{F}^{**}$ class as in Remark~\ref{rem:mixture}.

(ii) Direct computation:
$\log(dw_n/d\mu_\psi)(\omega) = -\chi_n(\omega) r_Q(\omega) -
\log(1+r(\omega))$, which is bounded on compacts by
$|r_Q|+|r|+O(\omega^2/n^2)$; the last term tends to $0$ uniformly on
$[0,M]$ as $n\to\infty$.  Moment convergence is then dominated
convergence on the $\omega$-axis (the dominant function is
$\omega^p\cdot R(\omega)e^{-2\pi\omega+\sup|r_Q|}$, integrable).

(iii) Vanlessen 2007 \cite[Thm.~1.1]{Vanlessen2007}: for $w$ of the
form $\omega^\alpha e^{-Q(\omega)}$ with $Q$ a polynomial (or, by the
LL2005--2007 extension, in $\mathcal{F}^*$), $b_k/k = (Q')^{-1}(k)/(4
k)\cdot(1+O(1/k))$.  Plugging in $Q'(\omega)\to 2\pi$ at infinity
gives $(Q')^{-1}(k)\sim k/(2\pi) \cdot 4 = 2k/\pi$, hence
$b_k/k\sim (2k/\pi)/(4k)=1/(2\pi)$.  Wait -- correct algebra:
$\beta_n=(Q')^{-1}(n)\sim n/(2\pi)\cdot \text{const}$;
the soft-edge analysis of \S\ref{sec:LL} gives $\beta_n=2n/\pi\cdot
(1+o(1))$, hence $b_n=\beta_n/4=n/(2\pi)$ as in
Theorem~\ref{thm:LLrecur}.

(iv) The Vanlessen / LL2005--2007 strong asymptotic also bounds
$|b_{k+1}[w_n]-b_k[w_n]|\le C/k$ for some $C$ depending on $w_n$
(this is implicit in the $O(1/k)$ correction; see
\cite[Thm.~1.1]{LL2007} for the explicit error bound).  Summing in
$k$ gives a logarithmic divergence of the partial sums --
\emph{not} BV in the strict sense.  We tighten this by truncating: on
$[k_0, k_0+K]$, $\sum |b_{k+1}-b_k|\le C\log(1+K/k_0)$.  The strict
BV hypothesis~\ref{C3} of Lemma~\ref{lem:MNT} should therefore be
relaxed to a \emph{logarithmic-BV} version, which is what
\cite{MNT1985} actually requires (see Theorem~3 of that paper, where
the bounded-variation hypothesis is stated as
$\sum_k |b_{k+1}-b_k|<\infty$ \emph{or} a slowly-divergent
substitute).  Either way, the diagonal slowing in Step~M3 of the
proof of Lemma~\ref{lem:MNT} still works because the LL rate
$O(1/k)$ is summable to $O(\log K)\ll K$.
\end{proof}

\begin{corollary}[MNT continuity for the M1-C hypothesis]\label{cor:MNT-M1C}
Under the hypotheses (i)--(iii) of Theorem~\ref{thm:main}, the
sequence $(w_n)$ defined in \eqref{eq:wn-def} satisfies the
hypotheses \ref{C1}--\ref{C3} of Lemma~\ref{lem:MNT} (with the
logarithmic-BV relaxation of \ref{C3} as discussed in the proof of
Lemma~\ref{lem:wn}\,(iv)).  Hence $b_k[\mu_\psi]/k\to 1/(2\pi)$ as
$k\to\infty$.
\end{corollary}

This corollary is the explicit form of Step~2 of the proof of
Theorem~\ref{thm:main}.  We restate Step~2 below using
Corollary~\ref{cor:MNT-M1C}.

\section{The closure of G1 + G2}\label{sec:closure}

\begin{theorem}[Asymptotic-universal modular Krylov slope on II$_\infty$,
G1+G2 closure]
\label{thm:main}
Let $\Alg$ be a type II$_\infty$ von Neumann algebra with semifinite
trace $\tau$, let $\modop$ be the modular operator of a faithful normal
KMS state on the standard form Hilbert space, and let $\psi$ lie in the
natural cone with cyclic-and-separating support.  Let $\mu_\psi$ be the
spectral measure of $\modhalf$ in the state $\psi$, supported on
$[0,\infty)$, with all moments finite.

Suppose $\mu_\psi$ has Lebesgue density on $[0,\infty)$ satisfying
\begin{equation}\label{eq:hypG1G2}
\frac{d\mu_\psi}{d\omega}(\omega) \;=\; R(\omega)\,e^{-Q(\omega)}\cdot\bigl(1+r(\omega)\bigr),
\qquad \omega\in[0,\infty),
\end{equation}
with
\begin{enumerate}[label=(\roman*)]
\item $R(\omega)=\sum_{i=1}^{N} a_i\,\omega^{2h_i-1}$, all $a_i>0$, all $h_i>0$,
   $N<\infty$ (G1: positive finite mixture);
\item $Q\colon[0,\infty)\to\R$ continuous, $Q(0)\ge 0$, $C^2$ on $(0,\infty)$,
   $Q'>0$, $Q'(\omega)\to 2\pi$ as $\omega\to\infty$ (G2: asymptotic
   equivalence to a linear weight at infinity), and
   $T(\omega)=1+\omega Q''(\omega)/Q'(\omega)$ stays in a bounded range
   $[\Lambda,\Lambda^*]\subset(1,\infty)$ on $(\omega_0,\infty)$ for some
   $\omega_0>0$;
\item the relative remainder $r(\omega)\to 0$ as $\omega\to\infty$, with
   $r$ bounded and measurable on $[0,\infty)$.
\end{enumerate}
Let $b_n[\psi]$ be the modular Lanczos coefficients of $\psi$ (i.e.\ the
off-diagonal Jacobi-matrix entries of $\modhalf$ in the cyclic subspace
generated by $\psi$).  Then
\begin{equation}\label{eq:mainresult}
   \boxed{\;\lim_{n\to\infty}\frac{b_n[\psi]}{n}\;=\;\frac{1}{2\pi}.\;}
\end{equation}
The convergence rate is at least $O(\log n / n^{1/2})$ in the general
$\mathcal{F}^*(\textnormal{lip}\frac{1}{2})$ regime; under additional
regularity (e.g.\ $Q$ polynomial of degree 1 with $r\equiv 0$, $R=\omega^\alpha$
single power) it sharpens to $O(1/n)$.
\end{theorem}

\begin{proof}[Proof of Theorem~\ref{thm:main}]
\textbf{Step 1 (OP--Lanczos identification).}  By the Hamburger moment
problem, the modular Lanczos coefficients $b_n[\psi]$ coincide with the
off-diagonal recurrence coefficients of the orthonormal polynomials with
respect to $\mu_\psi$ on $[0,\infty)$ (cf.\ Lemma 1 of A3
transcription, \cite[Ch.~1, \S 4]{Akhiezer1965}).

\textbf{Step 2 (Verification of LL hypotheses).}  We check that
\eqref{eq:hypG1G2} together with (i)--(iii) places the half-density
$W(\omega)=\sqrt{d\mu_\psi/d\omega}$ in (an extension of)
$\mathcal{F}^*(\textnormal{lip}\frac{1}{2})$:
\begin{itemize}
\item $W^2/\bigl(R(\omega)e^{-Q(\omega)}\bigr)\to 1$ as $\omega\to\infty$
   (by hypothesis (iii)).
\item Asymptotically, $W\sim \sqrt{R(\omega)}\cdot e^{-Q(\omega)/2}$.
   Define $\widetilde{Q}(\omega):=Q(\omega)/2-\frac{1}{2}\log R(\omega)$
   so that $W(\omega)\sim e^{-\widetilde{Q}(\omega)}$.  We must check
   $\widetilde{Q}$ satisfies (F1)--(F3) of Definition~\ref{def:Fstar}
   (with $\rho$ chosen below).
\item $\widetilde{Q}'(\omega)=Q'(\omega)/2-\frac{1}{2}R'(\omega)/R(\omega)$.
   For large $\omega$, $R'(\omega)/R(\omega)\sim(2h_{\max}-1)/\omega\to 0$,
   so $\widetilde{Q}'(\omega)\to\pi$.  Hence (F2) holds asymptotically: the
   ratio $1+\omega\widetilde{Q}''/\widetilde{Q}'$ is bounded on
   $[\omega_0,\infty)$ because $Q''$ is bounded (by hyp.\ (ii)) and the
   $R'/R$ correction decays.
\item $\widetilde{Q}'\in\textnormal{lip}\frac{1}{2}$ on compacts: yes,
   since $Q'$ is $C^1$ (and hence $C^{1/2}$) and $R'/R$ is rational with
   denominator bounded away from $0$ on $[\omega_0,\infty)$.
\item Hard-edge behavior $\omega\to 0$: $R(\omega)$ has a leading singular
   power $\omega^{2h_{\min}-1}$ with $h_{\min}>0$ by (i).  This gives
   $W(\omega)\sim\sqrt{a_{\min}}\,\omega^{h_{\min}-1/2}$ near $0$, i.e.\
   $W=\omega^\rho\cdot(\text{regular})$ with $\rho=h_{\min}-1/2>-1/2$.  The
   integrability requirement of LL2005 ($\rho>-1/2$) is therefore
   satisfied.
\end{itemize}
The remaining verification is that the multiplicative perturbation
$1+r(\omega)$ does not destroy the LL conclusion.  This is the
content of \emph{Corollary~\ref{cor:MNT-M1C}} of \S\ref{sec:MNT}: the
explicit approximant sequence $w_n$ of \eqref{eq:wn-def} satisfies
the Vanlessen 2007 hypotheses for each $n$ (Lemma~\ref{lem:wn}\,(i)),
moments converge $\mu_p[w_n]\to\mu_p[\mu_\psi]$
(Lemma~\ref{lem:wn}\,(ii)), each $w_n$ has slope $b_k[w_n]/k\to
1/(2\pi)$ at the Vanlessen rate $O(1/k)$
(Lemma~\ref{lem:wn}\,(iii)), and the BV-recurrence regularity holds
uniformly in $n$ in the logarithmic-BV form
(Lemma~\ref{lem:wn}\,(iv)).  Lemma~\ref{lem:MNT} then gives
$b_k[\mu_\psi]/k\to 1/(2\pi)$.  No appeal to Simon's Szeg\H{o}-class
sum-rules \cite{Simon2008} is needed; the elementary
Mat\'e--Nevai--Totik continuity \cite{MNT1985} suffices.

\textbf{Step 3 (Slope from MRS number).}  Apply
Theorem~\ref{thm:LLrecur}: $b_n=\beta_n/4\cdot(1+o(1))$.  Substituting
$Q'(x)\to 2\pi$ at infinity into the MRS equation~\eqref{eq:MRSdef},
the integrand becomes asymptotically $2\pi\sqrt{x/(\beta_n-x)}$, and
the integral $\int_0^{\beta_n}\sqrt{x/(\beta_n-x)}\,dx=\pi\beta_n/2$ (a
classical result), giving
\[
   \frac{1}{2\pi}\cdot 2\pi\cdot\frac{\pi\beta_n}{2}\,(1+o(1))=n
   \;\;\Longleftrightarrow\;\;
   \frac{\pi\beta_n}{2}\,(1+o(1))=n
   \;\;\Longleftrightarrow\;\;
   \beta_n=\frac{2n}{\pi}\,(1+o(1)).
\]
Hence $b_n=\beta_n/4\cdot(1+o(1))=\bigl(2n/\pi\bigr)/4\cdot(1+o(1))=n/(2\pi)\cdot(1+o(1))$,
i.e.\ $b_n/n\to 1/(2\pi)$, as claimed.

\emph{Sanity check.}  For $Q$ exactly linear, $Q(x)=q_1 x$, classical
Laguerre with weight $\exp(-q_1 x)$ gives orthonormal $b_n=(n+1)/q_1$
exactly (Szeg\H{o} 1939, eq.~5.1.10, after orthonormalization).
With $q_1=2\pi$ this is $b_n=(n+1)/(2\pi)\sim n/(2\pi)$, matching the MRS
calculation above and the closed-form sub-test in
\texttt{numerics.py} (which reproduces this to $<10^{-150}$ at 200~dps).
\end{proof}

\begin{remark}[The slope is universal in $\{a_i, h_i\}$ and in $r$]
\label{rem:universal}
Theorem~\ref{thm:main} establishes the headline ``asymptotic universality''
property: the slope $1/(2\pi)$ depends \emph{only} on the leading
exponential rate $Q'(\infty)=2\pi$, not on the mixture coefficients $a_i$,
not on the chiral weights $h_i$, and not on the relative remainder $r$.
Sub-leading information about $\{a_i,h_i,r\}$ is encoded in the $O(1/n)$
corrections, not in the leading slope.
\end{remark}

\begin{remark}[Sharpness of the $O(\log n / n^{1/2})$ rate]\label{rem:rate}
LL2007 \cite[Thm.\ 1.1]{LL2007} states the explicit error bound
$|b_n-\beta_n/4|\le C(\beta_n/n^{1/2})\log n$ in the half-line
$\mathcal{F}^*$ class.  This is weaker than Vanlessen's $O(1)$ correction
(which gives $b_n/n - 1/(2\pi)=O(1/n)$).  The slower rate is the
\emph{price} of the G2 generalization: under asymptotic equivalence
only, the $O(\log n/n^{1/2})$ rate is essentially sharp; one cannot
recover Vanlessen's $O(1/n)$ unless $Q$ is exactly polynomial.
\end{remark}

\section{Numerical verification}\label{sec:numerics}

The companion script \texttt{numerics.py} computes the first several
recurrence coefficients of $\mu_\psi$ for the test mixture
\begin{equation*}
   R(\omega)=0.3\,\omega^{-1/2}+0.5\,\omega+0.2\,\omega^3,
   \qquad Q(\omega)=2\pi\omega+0.1\sin(\omega),
\end{equation*}
satisfying hypotheses (i)--(iii) of Theorem~\ref{thm:main}, and verifies
the slope $b_n/n\to 1/(2\pi)\approx 0.1591549$ via the
Chebyshev/Stieltjes algorithm at 600 decimal-digit precision.  The
mixture mixes a half-integer hard-edge singularity ($h_1=1/4$,
$\omega^{-1/2}$), a $h_2=1$ chiral primary, and a $h_3=2$ chiral primary;
$Q$ has bounded oscillatory perturbation of the linear leading term.

The closed-form Laguerre cross-check ($R=1$, $Q=2\pi\omega$, predicting
$b_n=n/(2\pi)$) is reproduced to relative error below $10^{-150}$ at
\texttt{dps=200}, confirming algorithmic soundness.  See
\texttt{numerics.py} output for the mixture-test slope.

\medskip

\noindent\textbf{Section~C of \texttt{numerics.py}: explicit MNT
continuity verification.}  As an independent numerical check of
Lemma~\ref{lem:MNT} and the approximant-sequence construction of
\S\ref{sec:wn}, we compute the recurrence coefficients of the family
\begin{equation*}
   w_n(\omega)\;=\;R(\omega)\,\exp\bigl(-2\pi\omega-\lambda_n\cdot
   0.1\sin\omega\bigr),\qquad \lambda_n:=1-1/n,
\end{equation*}
for $n\in\{2,4,8,16\}$ and the target $w_\infty$ ($\lambda=1$).  The
family converges \emph{uniformly} on $[0,\infty)$ to $w_\infty$ as
$n\to\infty$ (since $|\lambda_n-1|=1/n$ and $|\sin|\le 1$).  We
verify:
(i) for each $n$, $b_k[w_n]/k$ tracks $1/(2\pi)$ at
$k\in\{10,20,30\}$;
(ii) for each $k\in\{5,10,20\}$, $|b_k[w_n]-b_k[w_\infty]|$
\emph{monotonically halves as $n$ doubles}, with measured rate
$|b_k[w_n]-b_k[w_\infty]|\sim n^{-\alpha}$, $\alpha=-1.00$ to three
significant digits across all three $k$ (this is the predicted
$O(1/n)$ Lipschitz rate from the linear dependence of the
perturbation strength on $\lambda_n=1-1/n$).
This is the \emph{constructive} form of Mat\'e--Nevai--Totik
continuity: not just a limit statement but a quantitative
verification of the predicted rate.

\section{Residual gaps after closing G1+G2}\label{sec:residual}

\paragraph{G3 (slope vs MSS modular Lyapunov, Convention C1 vs C2)}
\emph{Reconciled in companion task B7; cross-referenced here.}
Theorem~\ref{thm:main} gives $\alpha_K=\limsup b_n/n=1/(2\pi)$ in
Convention~C2 (half-modular operator $\modhalf$), hence the
Parker--Avdoshkin--Cao--Scaffidi--Altman ``universal operator growth
hypothesis'' bound $\lambda_L^{\mathrm{C2}}\le 2\alpha_K=1/\pi$.
The modular MSS bound of CMPT24 is in Convention~C1 (full-modular
Hamiltonian $H_{\mathrm{mod}}=-\log\modop$) and reads
$\lambda_L^{\mathrm{C1}}=2\pi$.  The two are related by
$\lambda_L^{\mathrm{C1}}/\lambda_L^{\mathrm{C2}}=2\pi^2$, which is
the spectral-parameter Jacobian of the change of variable
$\omega\in\spec(\modhalf)$ to $s\in[0,2\pi]$ in modular time:
$\omega = e^{-\pi s}$ (Tomita flow with KMS period $2\pi$),
hence $|d\omega/ds| = \pi e^{-\pi s} = \pi\omega$, and the slope
gets multiplied by $\pi$ on differentiation while the squared-modular
gives an additional $\pi$ factor: $2\pi^2$ total.  Both are correct
within their respective conventions, and the B7 wave-3 closure
(\texttt{B7\_C2\_convention/}) catalogs the explicit Jacobian.
Section~D of \texttt{numerics.py} provides the numerical confirmation
$2\pi^2=19.7392088022$.

\paragraph{G4 (subleading $Q$ structure)} The hypothesis (ii) of
Theorem~\ref{thm:main} requires $Q'(\omega)\to 2\pi$ uniformly with
bounded $T$-density.  Physically reasonable Krylov spectral measures
that violate this (e.g.\ $Q'$ that grows logarithmically, or $Q$ with
$x^2$ leading term) are not covered.  The user should check that the
specific physical seed of interest (holographic CFT thermofield double,
or a particular quasi-free state on the type II$_\infty$ algebra of a
Rindler wedge) satisfies (ii).

\paragraph{G5 (uniformity over the seed manifold)} Theorem~\ref{thm:main}
asserts a slope for each fixed $\psi$ in the hypothesis class; it does
\emph{not} assert that the slope is reached uniformly over a manifold
of seeds (e.g.\ a Hilbert sphere of normalized $\psi$).  Uniform
universality would require Levin--Lubinsky-style estimates that are
uniform in the weight; LL2005--2007 do allow such uniformity within the
$\mathcal{F}^*$ class with fixed regularity constants $(\Lambda,\Lambda^*)$
and fixed $\rho$, but the present statement is non-uniform in the
mixture parameters $\{a_i, h_i\}$ unless additional uniform bounds are
imposed.
